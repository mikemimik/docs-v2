```ts Example
export interface MfaPhoneEnrollmentMembers extends BaseMembers {
  client: ClientMembers;
  organization: OrganizationMembers;
  prompt: PromptMembers;
  screen: ScreenMembers;
  transaction: TransactionMembers;

  /**
   * Sélectionne l'indicatif de pays pour le numéro de téléphone.
   * @param payload Options personnalisées facultatives à inclure dans la requête.
   */
  pickCountryCode(payload?: CustomOptions): Promise<void>;

  /**
   * Poursuit le processus d'enrôlement avec le numéro de téléphone et le type fournis (SMS ou appel vocal).
   * @param payload Le numéro de téléphone, le type et les options personnalisées facultatives.
   */
  continueEnrollment(payload: ContinueOptions): Promise<void>;

  /**
   * Permet à l'utilisateur d'essayer une autre méthode de MFA.
   * @param payload Options personnalisées facultatives à inclure dans la requête.
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;
}
```
