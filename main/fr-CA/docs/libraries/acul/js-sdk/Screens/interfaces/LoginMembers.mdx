---
title: "LoginMembers"
---

Interface des membres de l’écran de connexion qui étend l’interface de membres de base

```ts Example
export interface LoginMembers extends BaseMembers {
  screen: ScreenMembersOnLogin;
  transaction: TransactionMembersOnLogin;
  /**
   * Effectue la connexion avec nom d'utilisateur/mot de passe
   * @param payload Les options de connexion
   */
  login(payload: LoginOptions): Promise<void>;
  /**
   * Effectue la connexion avec un fournisseur social
   * @param payload Les options de connexion sociale
   */
  federatedLogin(payload: FederatedLoginOptions): Promise<void>;
  /**
   * Sélectionne l'indicatif de pays pour la saisie du numéro de téléphone
   * @param payload Options personnalisées facultatives
   */
  pickCountryCode(payload?: CustomOptions): Promise<void>;
  /**
   * Récupère les types d'identifiant actifs pour l'écran de connexion
   * @returns Un tableau des types d'identifiant actifs ou null si aucun n'est actif
   * @utilityFeature
   */
  getLoginIdentifiers(): IdentifierType[] | null;
}
```
