ConsentMembers

````ts Example
export interface ConsentMembers extends BaseMembers {
  /**
   * Fournit l'accès à la propriété et donnée propres de l'écran de consentement,
   * y compris la liste des `scopes` demandés et le flag `hideScopes`.
   * @type {ScreenMembersOnConsent}
   */
  screen: ScreenMembersOnConsent;

  /**
   * Soumet la décision de l'utilisateur d'accepter (accorder) les permissions demandées.
   * Cette action envoie une requête POST à l'endpoint `/u/consent` avec `action: "accept"`.
   *
   * @param {CustomOptions} [payload] - Facultatif. Un objet pour toute paire clé-valeur personnalisée
   *                                    à envoyer avec la requête. Ces paramètres seront
   *                                    inclus dans les données du formulaire soumis au serveur.
   * @returns {Promise<void>} Une Promise qui se résout lorsque l'action d'acceptation est soumise avec succès.
   *                          En cas de succès, Auth0 redirige généralement l'utilisateur vers l'application
   *                          ou vers l'étape suivante du flux d'authentification.
   * @throws {Error} Génère une erreur si la soumission du formulaire échoue (p. ex., problème réseau, état invalide).
   *                 Les erreurs côté serveur (comme « invalid_request ») seront reflétées dans `this.transaction.errors`
   *                 après l'opération, plutôt que d'être générées comme des erreurs JavaScript.
   *
   * @example
   * ```typescript
   * // En supposant que 'consentManager' est une instance de la classe SDK de l'écran de consentement
   * try {
   *   await consentManager.accept();
   *   // En cas de succès, la page sera généralement redirigée.
   * } catch (error) {
   *   // Gérer les erreurs inattendues lors de la soumission elle-même.
   *   console.error("Failed to submit consent acceptance:", error);
   * }
   * // Après l'attente, vérifier consentManager.transaction.errors pour les problèmes de validation côté serveur.
   * ```
   */
  accept(payload?: CustomOptions): Promise<void>;

  /**
   * Soumet la décision de l'utilisateur de rejeter les permissions demandées.
   * Cette action envoie une requête POST à l'endpoint `/u/consent` avec `action: "deny"`.
   *
   * @param {CustomOptions} [payload] - Facultatif. Un objet pour toute paire clé-valeur personnalisée
   *                                    à envoyer avec la requête. Ces paramètres seront
   *                                    inclus dans les données du formulaire soumis au serveur.
   * @returns {Promise<void>} Une Promise qui se résout lorsque l'action de rejet est soumise avec succès.
   *                          En cas de succès, Auth0 redirige généralement l'utilisateur, en affichant
   *                          potentiellement un message d'accès refusé ou en retournant une erreur à l'application.
   * @throws {Error} Génère une erreur si la soumission du formulaire échoue (p. ex., problème réseau, état invalide).
   *                 Les erreurs côté serveur sont reflétées dans `this.transaction.errors`.
   *
   * @example
   * ```typescript
   * // En supposant que 'consentManager' est une instance de la classe SDK de l'écran de consentement
   * try {
   *   await consentManager.deny({ reason: "user_declined" }); // Exemple d'option personnalisée
   *   // En cas de succès, la page sera généralement redirigée.
   * } catch (error) {
   *   console.error("Failed to submit consent denial:", error);
   * }
   * // Après l'attente, vérifier consentManager.transaction.errors pour les problèmes de validation côté serveur.
   * ```
   */
  deny(payload?: CustomOptions): Promise<void>;
}

<ParamField body="accept" type="Promise<void>">
  Soumet la décision de l&#39;utilisateur d&#39;accepter (d&#39;accorder) les permissions demandées.
  Cette action envoie une requête POST au point de terminaison `/u/consent` avec `action: "accept"`.

  Une promesse qui est résolue lorsque l&#39;action d&#39;acceptation est soumise avec succès.
  En cas de réussite, Auth0 redirige généralement l&#39;utilisateur vers l&#39;application
  ou vers l&#39;étape suivante du flux d&#39;authentification.

  #### Throws

  Lève une erreur si la soumission du formulaire échoue (p. ex., problème réseau, invalid state).
  Les erreurs côté serveur (comme &quot;invalid&#95;request&quot;) se reflètent dans `this.transaction.errors`
  après l&#39;opération, plutôt que d&#39;être levées comme des erreurs JavaScript.

  ```typescript Example
  // En supposant que 'consentManager' est une instance de la classe SDK de l'écran de consentement
  try {
    await consentManager.accept();
    // En cas de réussite, la page sera généralement redirigée.
  } catch (error) {
    // Gérer les erreurs inattendues pendant la soumission elle-même.
    console.error("Failed to submit consent acceptance:", error);
  }
  // Après le await, vérifiez consentManager.transaction.errors pour les problèmes de validation côté serveur.
  ```
