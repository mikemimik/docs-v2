export interface ScreenMembersOnResetPasswordMfaWebAuthnRoamingChallenge extends ScreenMembers {
  /**
   * Un accesseur direct pour les données `passkey.public_key` provenant de `screen.data`.
   * Cet objet contient le défi et les autres options nécessaires à l'appel d'API
   * `navigator.credentials.get()` de WebAuthn API.
   * La valeur est `null` si `screen.data.passkey.public_key` n'est pas disponible.
   * @type {PasskeyRead['public_key'] | null}
   * @public
   */
  publicKey: PasskeyRead['public_key'] | null;

  /**
   * Un accesseur direct pour l'indicateur `show_remember_device` provenant de `screen.data`.
   * Indique si l'interface utilisateur doit présenter à l'utilisateur une option pour mémoriser cet appareil/navigateur
   * afin de potentiellement contourner l'AMF lors des connexions futures. La valeur par défaut est `false` si non défini dans `screen.data`.
   * @type {boolean}
   * @public
   */
  showRememberDevice: boolean;
}
