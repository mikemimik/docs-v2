export interface ContinueOptions extends CustomOptions {
  /**
   * Le mode d’envoi sélectionné par l'utilisateur pour recevoir le code de vérification.
   * - `sms`: Envoyer le code par message texte.
   * - `voice`: Envoyer le code par appel vocal.
   * @type {'sms' | 'voice'}
   */

}
