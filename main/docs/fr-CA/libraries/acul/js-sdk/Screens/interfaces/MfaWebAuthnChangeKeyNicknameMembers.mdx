export interface MfaWebAuthnChangeKeyNicknameMembers extends BaseMembers {
  /**
   * L'objet screen avec la structure de données spécifique à MFA WebAuthn Change Key Nickname.
   */
  screen: ScreenMembersOnMfaWebAuthnChangeKeyNickname;
  
  /**
   * Soumet le nouveau surnom pour la clé de sécurité WebAuthn.
   * Cette action correspond à l'utilisateur qui saisit un nouveau surnom et clique sur un bouton « Sauvegarder » ou « Continuer ».
   * Le SDK enverra ce nouveau surnom via une requête POST au point de terminaison Auth0 `/u/mfa-webauthn-change-key-nickname`.
   *
   * Si le nouveau surnom est valide et que la mise à jour réussit, Auth0 redirigera généralement l'utilisateur
   * vers l'écran suivant approprié (p. ex., retour à la gestion des facteurs MFA ou une confirmation de succès).
   * Si le surnom est invalide (p. ex., vide, trop long, trop court), Auth0 effectuera un nouveau rendu de l'écran
   * 'mfa-webauthn-change-key-nickname', et le tableau `transaction.errors` dans le contexte du SDK
   * sera mis à jour avec les détails de l'échec de validation.
   *
   * @param {ContinueOptions} payload - Un objet contenant le `nickname` (string) saisi par l'utilisateur.
   *                                    Il peut également inclure tout `CustomOptions` pour l'extensibilité.
   * @returns {Promise<void>} Une promesse qui se résout lorsque la soumission du formulaire est lancée.
   *                          Elle ne retourne pas de données directement à la résolution, car une redirection ou
   *                          un nouveau rendu de la page est le résultat habituel.
   * @throws {Error} Génère une erreur si `payload.nickname` n'est pas fourni ou n'est pas un string,
   *                 ou si le `FormHandler` rencontre un problème irrécupérable lors de la soumission (p. ex., erreur réseau).
   *                 Les erreurs de validation provenant d'Auth0 (comme un surnom invalide) ne sont pas levées en tant qu'erreurs JavaScript,
   *                 mais sont reflétées dans `this.transaction.errors` après l'opération.
   *
   * @example
   * ```typescript
   * // En supposant que 'sdk' est une instance de MfaWebAuthnChangeKeyNickname
   * const newNickname = "My Favorite YubiKey"; // Value from user input
   * try {
   *   await sdk.continueWithNewNickname({ nickname: newNickname });
   *   // En cas de succès, la page effectue une redirection.
   * } catch (error) {
   *   // Ce bloc catch est destiné aux erreurs inattendues lors de la requête SDK elle-même.
   *   console.error("Failed to submit the new nickname:", error);
   * }
   * // Après l'await, vérifiez toujours sdk.transaction.errors pour les problèmes de validation côté serveur.
   * if (sdk.transaction.errors && sdk.transaction.errors.length > 0) {
   *   sdk.transaction.errors.forEach(err => {
   *     if (err.field === 'nickname') {
   *       // Afficher err.message lié au champ de saisie du surnom.
   *       // p. ex., "Name is too short", "Name is required"
   *     }
   *   });
   * }
   * ```
   */
  continueWithNewNickname(payload: ContinueOptions): Promise<void>;
}

<ParamField body="continueWithNewNickname" type="Promise<void>">
  Soumet le nouveau surnom de la clé de sécurité WebAuthn.
  Cette action correspond au moment où l’utilisateur saisit un nouveau surnom et clique sur un bouton « Save » ou « Continue ».
  Le SDK enverra par POST ce nouveau surnom au point de terminaison Auth0 `/u/mfa-webauthn-change-key-nickname`.

  Si le nouveau surnom est valide et que la mise à jour réussit, Auth0 redirigera généralement l’utilisateur
  vers l’écran suivant approprié (p. ex., de retour à la gestion des facteurs MFA ou vers une confirmation de réussite).
  Si le surnom n’est pas valide (p. ex., vide, trop long, trop court), Auth0 affichera de nouveau l’écran
  &#39;mfa-webauthn-change-key-nickname&#39;, et le tableau `transaction.errors` dans le contexte du SDK
  sera mis à jour avec les détails de l’erreur de validation.

  Une promesse qui se résout lorsque la soumission du formulaire est lancée.
  Elle ne retourne pas directement de données lors de sa résolution, puisqu’une redirection ou un
  réaffichage de la page est le résultat habituel.

  #### Throws

  Lève une erreur si `payload.nickname` n’est pas fourni ou n’est pas une chaîne de caractères,
  ou si le `FormHandler` rencontre un problème irrécupérable pendant la soumission (p. ex., une erreur réseau).
  Les erreurs de validation provenant d’Auth0 (comme un surnom invalide) ne sont pas levées comme erreurs JavaScript,
  mais se reflètent dans `this.transaction.errors` après l’opération.

  ```typescript Example
  // En supposant que 'sdk' est une instance de MfaWebAuthnChangeKeyNickname
  const newNickname = "My Favorite YubiKey"; // Valeur provenant de la saisie de l’utilisateur
  try {
    await sdk.continueWithNewNickname({ nickname: newNickname });
    // En cas de réussite, la page redirige.
  } catch (error) {
    // Ce catch sert aux erreurs inattendues pendant l’appel du SDK lui-même.
    console.error("Failed to submit the new nickname:", error);
  }
  // Après le await, vérifiez toujours sdk.transaction.errors pour les problèmes de validation côté serveur.
  if (sdk.transaction.errors && sdk.transaction.errors.length > 0) {
    sdk.transaction.errors.forEach(err => {
      if (err.field === 'nickname') {
        // Affichez err.message lié au champ de saisie du surnom.
        // p. ex., "Name is too short", "Name is required"
      }
    });
  }
  ```
