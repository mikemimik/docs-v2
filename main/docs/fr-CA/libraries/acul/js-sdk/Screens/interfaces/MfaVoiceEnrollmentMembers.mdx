---
title: "MfaVoiceEnrollmentMembers"
---

Interface décrivant les membres de l’écran d’inscription vocale MFA.

```ts Example
export interface MfaVoiceEnrollmentMembers extends BaseMembers {
  client: ClientMembers;
  organization: OrganizationMembers;
  prompt: PromptMembers;
  screen: ScreenMembers;
  transaction: TransactionMembers;

  /**
   * Poursuit le processus d'inscription vocale MFA.
   * @param payload - Le numéro de téléphone et les options personnalisées facultatives.
   * @returns Promise qui se résout lorsque l'inscription se poursuit.
   */
  continue(payload: ContinueOptions): Promise<void>;

  /**
   * Permet d'essayer une autre méthode d'authentification.
   * @param payload - Options personnalisées facultatives.
   * @returns Promise qui se résout lorsque l'utilisateur change de méthode.
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;

  /**
   * Permet de sélectionner un indicatif de pays pour le numéro de téléphone.
   * @param payload - Options personnalisées facultatives.
   * @returns Promise qui se résout lorsque l'indicatif de pays est sélectionné.
   */
  selectPhoneCountryCode(payload?: CustomOptions): Promise<void>;
}
```
