export interface MfaWebAuthnNotAvailableErrorMembers extends BaseMembers {
  /**
   * Permet à l'utilisateur de sélectionner une méthode AMF différente parce que WebAuthn n'est pas disponible.
   * Cette action dirige l'utilisateur vers un écran de sélection d'authentificateur où il peut choisir
   * parmi d'autres facteurs AMF disponibles ou inscrits.
   *
   * @param {CustomOptions} [payload] - Paramètres personnalisés optionnels à envoyer avec la requête.
   * @returns {Promise<void>} Une promesse qui se résout lors de la soumission réussie de l'action 'pick-authenticator'.
   * @throws {Error} Génère une erreur si la soumission du formulaire échoue (p. ex., problèmes réseau, erreurs de validation côté serveur).
   * @example
   * ```typescript
   * // En supposant que 'sdk' est une instance de MfaWebAuthnNotAvailableError
   * try {
   *   await sdk.tryAnotherMethod();
   *   // En cas de succès, Auth0 gère la redirection vers l'écran de sélection d'authentificateur.
   * } catch (error) {
   *   console.error('Failed to switch to another MFA method:', error);
   *   // Mettre à jour l'interface utilisateur au besoin pour informer l'utilisateur de l'échec.
   * }
   * ```
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;
}

<ParamField body="tryAnotherMethod" type="Promise<void>">
  Permet à l’utilisateur de choisir une autre méthode d’AMF lorsque WebAuthn n’est pas disponible.
  Cette action redirige l’utilisateur vers un écran de sélection de l’authentificateur, où il peut choisir
  parmi les autres facteurs d’AMF disponibles ou déjà inscrits.

  Une promesse qui est résolue lorsque la soumission de l’action &#39;pick-authenticator&#39; réussit.

  #### Lance

  Lance une erreur si la soumission du formulaire échoue (p. ex., en cas de problème réseau ou d’erreurs de validation côté serveur).

  ```typescript Example
  // En supposant que 'sdk' est une instance de MfaWebAuthnNotAvailableError
  try {
    await sdk.tryAnotherMethod();
    // En cas de réussite, Auth0 gère la redirection vers l'écran de sélection de l'authentificateur.
  } catch (error) {
    console.error('Failed to switch to another MFA method:', error);
    // Mettre à jour l’UI au besoin pour informer l’utilisateur de l’échec.
  }
  ```
