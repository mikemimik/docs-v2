---
title: "ResetPasswordMfaVoiceChallengeMembers"
---

Interface qui définit les méthodes et les propriétés disponibles pour l’écran reset-password-mfa-voice-challenge.

````ts Example
export interface ResetPasswordMfaVoiceChallengeMembers extends BaseMembers {
  screen: ScreenMembersOnResetPasswordMfaVoiceChallenge;

  /**
   * Poursuit le défi vocal à l'aide du code fourni.
   *
   * @param payload - Les options contenant le code.
   * @returns Une promesse qui se résout lorsque le défi est soumis.
   *
   * @example
   * ```ts
   * const reset = new ResetPasswordMfaVoiceChallenge();
   * await reset.continue({ code: '123456' });
   * ```
   */
  continue(payload: ContinueOptions): Promise<void>;

  /**
   * Bascule vers la vérification par SMS.
   *
   * @param payload - Options personnalisées facultatives à inclure dans la requête.
   * @returns Une promesse qui se résout lorsque l'action est terminée.
   */
  switchToSms(payload?: CustomOptions): Promise<void>;

  /**
   * Renvoie le code par appel vocal.
   *
   * @param payload - Options personnalisées facultatives à inclure dans la requête.
   * @returns Une promesse qui se résout lorsque le code est renvoyé.
   */
  resendCode(payload?: CustomOptions): Promise<void>;

  /**
   * Permet à l'utilisateur d'essayer une autre méthode d'AMF.
   *
   * @param payload - Options personnalisées facultatives à inclure dans la requête.
   * @returns Une promesse qui se résout lorsque la requête est soumise.
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;

  /**
   * Obtient la fonctionnalité de renvoi avec gestion du délai d'attente pour cet écran
   * @param options Options de configuration pour la fonctionnalité de renvoi
   */
  resendManager(options?: StartResendOptions): ResendControl;
}
````
