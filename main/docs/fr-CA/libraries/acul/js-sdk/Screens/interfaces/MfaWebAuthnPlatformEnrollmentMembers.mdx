export interface MfaWebAuthnPlatformEnrollmentMembers extends BaseMembers {
  /**
   * L'objet screen avec la structure de données spécifique à MFA WebAuthn Platform Enrollment.
   */
  screen: ScreenMembersOnMfaWebAuthnPlatformEnrollment;
  
  /**
   * Lance le processus de création de information d’identification WebAuthn platform en utilisant les option de clé publique
   * disponibles sur `this.screen.publicKey` et soumet le information d’identification WebAuthn résultant au serveur.
   * Cette méthode appelle en interne `createPasskeyCredentials` (qui encapsule `navigator.credentials.create()`).
   *
   * @param {SubmitPasskeyCredentialOptions} [payload] - Paramètres personnalisés optionnels à envoyer au serveur
   * avec le information d’identification WebAuthn créé.
   * @returns {Promise<void>} Une promesse qui se résout lorsque le information d’identification WebAuthn est créé et soumis avec succès.
   * @throws {Error} Génère une erreur si `this.screen.publicKey` n'est pas disponible, si `createPasskeyCredentials` échoue
   * (p. ex., annulation par l'utilisateur, problèmes matériels), ou si la soumission au serveur échoue.
   * @example
   * ```typescript
   * // En supposant que 'sdk' est une instance de MfaWebAuthnPlatformEnrollment
   * try {
   *   if (!sdk.screen.publicKey) { // Vérifier si les options sont disponibles
   *     throw new Error("Public key creation options are not available on the screen context.");
   *   }
   *   await sdk.submitPasskeyCredential(); // Inutile de passer publicKey explicitement
   *   // En cas de succès, Auth0 gère la redirection.
   * } catch (error) {
   *   console.error('Passkey enrollment failed:', error);
   *   // Gérer l'erreur, en appelant reportBrowserError si nécessaire lorsqu'il s'agit d'une WebAuthn API error
   *   if (error.name && error.message) { // Vérifier si cela ressemble à une WebAuthn error
   *     await sdk.reportBrowserError({ error: { name: error.name, message: error.message } });
   *   }
   * }
   * ```
   */
  submitPasskeyCredential(payload?: SubmitPasskeyCredentialOptions): Promise<void>;

   * }

<ParamField body="submitPasskeyCredential" type="Promise<void>">
  Lance le processus de création des informations d’identification WebAuthn de plateforme à l’aide des options de clé publique
  disponibles dans `this.screen.publicKey`, puis soumet l’information d’identification WebAuthn obtenue au serveur.
  Cette méthode appelle en interne `createPasskeyCredentials` (qui encapsule `navigator.credentials.create()`).

  Une promesse qui se résout lorsque l’information d’identification WebAuthn est créée et soumise avec succès.

  #### Exceptions

  Lève une erreur si `this.screen.publicKey` n’est pas disponible, si `createPasskeyCredentials` échoue
  (par ex., annulation par l’utilisateur, problèmes matériels) ou si la soumission au serveur échoue.

  ```typescript Example
  // En supposant que 'sdk' est une instance de MfaWebAuthnPlatformEnrollment
  try {
    if (!sdk.screen.publicKey) { // Vérifiez si les options sont disponibles
      throw new Error("Public key creation options are not available on the screen context.");
    }
    await sdk.submitPasskeyCredential(); // Il n’est pas nécessaire de transmettre publicKey explicitement
    // En cas de succès, Auth0 gère la redirection.
  } catch (error) {
    console.error('Passkey enrollment failed:', error);
    // Gérez l’erreur, potentiellement en appelant reportBrowserError s’il s’agit d’une erreur de l’API WebAuthn
    if (error.name && error.message) { // Vérifiez si cela ressemble à une erreur WebAuthn
      await sdk.reportBrowserError({ error: { name: error.name, message: error.message } });
    }
  }
  ```
