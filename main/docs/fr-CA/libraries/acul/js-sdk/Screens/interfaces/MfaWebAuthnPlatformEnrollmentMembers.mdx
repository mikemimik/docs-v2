export interface MfaWebAuthnPlatformEnrollmentMembers extends BaseMembers {
  /**
   * The screen object with MFA WebAuthn Platform Enrollment specific data structure.

   * Initiates the WebAuthn platform identifiant creation process using the public key options
   * available on `this.screen.publicKey` and submits the resulting identifiant to the server.
   * This method internally calls `createPasskeyCredentials` (which wraps `navigator.credentials.create()`).
   *
   * @param {SubmitPasskeyCredentialOptions} [payload] - Optional custom parameters to be sent to the server
   * along with the created identifiant.
   * @returns {Promise<void>} A promise that resolves when the identifiant is successfully created and submitted.

   *   }
   *   await sdk.submitPasskeyCredential(); // No need to pass publicKey explicitly
   *   // On success, Auth0 handles redirection.
   * } catch (error) {
   *   console.error('Clé d’accès enrollment failed:', error);

   *   if (error.name && error.message) { // Check if it looks like a WebAuthn error
   *     await sdk.reportBrowserError({ error: { name: error.name, message: error.message } });
   *   }
   * }
   * ```
   */

<ParamField body="submitPasskeyCredential" type="Promise<void>">
  Lance le processus de création de l’identifiant WebAuthn de plateforme à l’aide des options de clé publique
  disponibles dans `this.screen.publicKey` et soumet l’identifiant obtenu au serveur.
  Cette méthode appelle en interne `createPasskeyCredentials` (qui encapsule `navigator.credentials.create()`).

  Une promesse résolue lorsque l’identifiant est créé et soumis avec succès.

  #### Exceptions

  Génère une erreur si `this.screen.publicKey` n’est pas disponible, si `createPasskeyCredentials` échoue
  (par ex., annulation par l’utilisateur, problèmes matériels), ou si la soumission au serveur échoue.

  ```typescript Example
  // En supposant que 'sdk' est une instance de MfaWebAuthnPlatformEnrollment
  try {
    if (!sdk.screen.publicKey) { // Vérifier si les options sont disponibles
      throw new Error("Public key creation options are not available on the screen context.");
    }
    await sdk.submitPasskeyCredential(); // Pas besoin de transmettre publicKey explicitement
    // En cas de succès, Auth0 gère la redirection.
  } catch (error) {
    console.error('Passkey enrollment failed:', error);
    // Gérer l’erreur, potentiellement en appelant reportBrowserError s’il s’agit d’une erreur de l’API WebAuthn
    if (error.name && error.message) { // Vérifier si cela ressemble à une erreur WebAuthn
      await sdk.reportBrowserError({ error: { name: error.name, message: error.message } });
    }
  }
  ```
