```ts Example
export interface ShowErrorOptions {
  /**
   * L'objet d'erreur capturé lors d'un appel échoué à `navigator.credentials.get()`.
   * Celui-ci doit inclure au moins les propriétés `name` et `message` du DOMException.
   * @type {WebAuthnErrorDetails}
   * @example { name: "NotAllowedError", message: "The operation was aborted by the user." }
   */
  error: WebAuthnErrorDetails;

  /**
   * Facultatif. Si `true`, et si le contexte d'écran (`screen.showRememberDevice`)
   * indique que la mémorisation de l'appareil est une option, cela signale l'intention
   * de l'utilisateur de mémoriser ce navigateur/appareil.
   * @type {boolean | undefined}
   */
  rememberDevice?: boolean;
}
```

<ParamField body="error" type={<span><a href="/docs/fr-CA/libraries/acul/react-sdk/API-Reference/Types/interfaces/WebAuthnErrorDetails">WebAuthnErrorDetails</a></span>}>
  L’objet d’erreur renvoyé par l’API WebAuthn (`navigator.credentials.get()`),
  qui contient les valeurs `name` et `message` de la DOMException.
</ParamField>

<ParamField body="rememberDevice?" type="boolean">
  Facultatif. Si `true` et que `screen.showRememberDevice` est aussi `true`,
  le SDK inclura `rememberBrowser=true` dans la soumission du formulaire.
</ParamField>
