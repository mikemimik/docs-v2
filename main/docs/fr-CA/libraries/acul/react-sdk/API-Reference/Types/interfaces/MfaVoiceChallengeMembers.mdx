---
title: "MfaVoiceChallengeMembers"
---

Interface qui définit tous les membres et toutes les opérations offerts par l’écran de défi vocal de l’AMF.

````ts Example
export interface MfaVoiceChallengeMembers extends BaseMembers {
  /**
   * Propriété et donnée propre à l'écran.
   */
  screen: ScreenMembersOnMfaVoiceChallenge;

  untrustedData: UntrustedDataMembersOnMfaVoiceChallenge;

  /**
   * Soumet le code de vérification vocal pour valider le défi AMF.
   *
   * @param payload - Objet contenant le code de vérification et les paramètres facultatifs
   * @returns Promise qui se résout lorsque le code est validé avec succès
   *
   * @example
   * ```typescript
   * const mfaVoiceChallenge = new MfaVoiceChallenge();
   * mfaVoiceChallenge.continue({
   *   code: '123456',
   *   rememberDevice: true
   * });
   * ```
   */
  continue(payload: MfaVoiceChallengeContinueOptions): Promise<void>;

  /**
   * Navigue vers l'écran de sélection d'un autre numéro de téléphone.
   *
   * @param payload - Paramètres personnalisés facultatifs
   * @returns Promise qui se résout lorsque la navigation est terminée
   *
   * @example
   * ```typescript
   * const mfaVoiceChallenge = new MfaVoiceChallenge();
   * mfaVoiceChallenge.pickPhone();
   * ```
   */
  pickPhone(payload?: CustomOptions): Promise<void>;

  /**
   * Bascule vers la méthode de vérification par SMS plutôt que par appel vocal.
   *
   * @param payload - Paramètres personnalisés facultatifs
   * @returns Promise qui se résout lorsque le basculement est terminé
   *
   * @example
   * ```typescript
   * const mfaVoiceChallenge = new MfaVoiceChallenge();
   * mfaVoiceChallenge.switchToSms();
   * ```
   */
  switchToSms(payload?: CustomOptions): Promise<void>;

  /**
   * Demande un nouvel appel vocal avec un code de vérification.
   *
   * @param payload - Paramètres personnalisés facultatifs
   * @returns Promise qui se résout lorsque le nouveau code est envoyé
   *
   * @example
   * ```typescript
   * const mfaVoiceChallenge = new MfaVoiceChallenge();
   * mfaVoiceChallenge.resendCode();
   * ```
   */
  resendCode(payload?: CustomOptions): Promise<void>;

  /**
   * Navigue vers l'écran de sélection d'une autre méthode AMF.
   *
   * @param payload - Paramètres personnalisés facultatifs
   * @returns Promise qui se résout lorsque la navigation est terminée
   *
   * @example
   * ```typescript
   * const mfaVoiceChallenge = new MfaVoiceChallenge();
   * mfaVoiceChallenge.tryAnotherMethod();
   * ```
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;

  /**
   * Obtient la fonctionnalité de renvoi avec gestion du délai d'attente pour cet écran
   * @param options Option de configuration pour la fonctionnalité de renvoi
   */
  resendManager(options?: StartResendOptions): ResendControl;
}
````
