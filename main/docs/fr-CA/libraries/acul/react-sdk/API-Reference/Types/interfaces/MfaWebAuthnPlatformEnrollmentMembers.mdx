export interface MfaWebAuthnPlatformEnrollmentMembers extends BaseMembers {
  /**
   * l’objet screen contenant la structure de données propre à MFA WebAuthn Platform Enrollment.
   */
  screen: ScreenMembersOnMfaWebAuthnPlatformEnrollment;
  
  /**
   * Initiates the WebAuthn platform credential creation process using the public key options
   * available on `this.screen.publicKey` and submits the resulting credential to the server.
   * This method internally calls `createPasskeyCredentials` (which wraps `navigator.credentials.create()`).
   *
   * @param {SubmitPasskeyCredentialOptions} [payload] - Paramètres personnalisés optionnels à envoyer au serveur
   * avec le credential créé.
   * @returns {Promise<void>} Une Promise qui se résout lorsque le credential est créé et soumis avec succès.
   * @throws {Error} Génère une erreur si `this.screen.publicKey` n'est pas disponible, si `createPasskeyCredentials` échoue
   * (p. ex., annulation par l'utilisateur, problèmes matériels), ou si la soumission au serveur échoue.
   * @example
   * ```typescript
   * // Assuming 'sdk' is an instance of MfaWebAuthnPlatformEnrollment
   * try {
   *   if (!sdk.screen.publicKey) { // Check if options are available
   *     throw new Error("Public key creation options are not available on the screen context.");
   *   }
   *   await sdk.submitPasskeyCredential(); // No need to pass publicKey explicitly
   *   // On success, Auth0 handles redirection.
   * } catch (error) {
   *   console.error('Passkey enrollment failed:', error);

   *   if (error.name && error.message) { // Check if it looks like a WebAuthn error
   *     await sdk.reportBrowserError({ error: { name: error.name, message: error.message } });
   *   }
   * }
   * ```
   */
  submitPasskeyCredential(payload?: SubmitPasskeyCredentialOptions): Promise<void>;

<ParamField body="submitPasskeyCredential" type="Promise<void>">
  Lance le processus de création d’un credential de plateforme WebAuthn à l’aide des options de clé publique
  disponibles dans `this.screen.publicKey`, puis soumet le credential obtenu au serveur.
  Cette méthode appelle en interne `createPasskeyCredentials` (qui encapsule `navigator.credentials.create()`).

  Une Promise qui se résout lorsque le credential est créé et soumis avec succès.

  #### Lance

  Lance une erreur si `this.screen.publicKey` n’est pas disponible, si `createPasskeyCredentials` échoue
  (par ex. annulation par l’utilisateur, problèmes matériels), ou si la soumission au serveur échoue.

  ```typescript Example
  // En supposant que 'sdk' est une instance de MfaWebAuthnPlatformEnrollment
  try {
    if (!sdk.screen.publicKey) { // Vérifiez si les options sont disponibles
      throw new Error("Public key creation options are not available on the screen context.");
    }
    await sdk.submitPasskeyCredential(); // Pas besoin de transmettre publicKey explicitement
    // En cas de succès, Auth0 gère la redirection.
  } catch (error) {
    console.error('Passkey enrollment failed:', error);
    // Gérer l'erreur, potentiellement en appelant reportBrowserError s'il s'agit d'une erreur de l'API WebAuthn
    if (error.name && error.message) { // Vérifiez si cela ressemble à une erreur WebAuthn
      await sdk.reportBrowserError({ error: { name: error.name, message: error.message } });
    }
  }
  ```
