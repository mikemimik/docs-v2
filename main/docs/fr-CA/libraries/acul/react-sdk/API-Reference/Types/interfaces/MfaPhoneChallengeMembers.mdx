MfaPhoneChallengeMembers

````ts Example
export interface MfaPhoneChallengeMembers extends BaseMembers {
  /**
   * @property {ScreenMembersOnMfaPhoneChallenge} screen - Informations spécifiques à l'écran pour le MFA Phone Challenge.
   */
  screen: ScreenMembersOnMfaPhoneChallenge;

  /**
   * continue
   * Soumet le choix de la méthode d’envoi de l’AMF (SMS ou Voix) pour poursuivre le défi.
   * Cela déclenchera l'envoi du code de vérification au téléphone de l'utilisateur.
   *
   * @param {ContinueOptions} payload - Les options précisant le type de livraison ('sms' ou 'voice').
   * @returns {Promise<void>} Une promesse qui se résout lorsque l'action est soumise.
   * @throws {Error} Génère une erreur si la soumission échoue.
   * @example
   * ```typescript
   * import MfaPhoneChallenge from '@auth0/auth0-acul-js/mfa-phone-challenge';
   *
   * const mfaPhoneChallenge = new MfaPhoneChallenge();
   *
   * // Pour envoyer le code par SMS
   * await mfaPhoneChallenge.continue({ type: 'sms' });
   *
   * // Pour envoyer le code par appel vocal
   * await mfaPhoneChallenge.continue({ type: 'voice' });
   * ```
   */
  continue(payload: ContinueOptions): Promise<void>;

  /**
   * pickPhone
   * Lance l'action permettant de choisir un numéro de téléphone différent.
   * Cela redirige généralement l'utilisateur vers un écran où il peut sélectionner parmi plusieurs numéros de téléphone inscrits, le cas échéant.
   *
   * @param {PickPhoneOptions} [payload] - Options personnalisées facultatives à inclure avec la requête.
   * @returns {Promise<void>} Une promesse qui se résout lorsque l'action est soumise.
   * @throws {Error} Génère une erreur si la soumission échoue.
   * @example
   * ```typescript
   * import MfaPhoneChallenge from '@auth0/auth0-acul-js/mfa-phone-challenge';
   *
   * const mfaPhoneChallenge = new MfaPhoneChallenge();
   * await mfaPhoneChallenge.pickPhone();
   * ```
   */
  pickPhone(payload?: PickPhoneOptions): Promise<void>;

  /**
   * tryAnotherMethod
   * Lance l'action permettant de sélectionner un facteur AMF/authentificateur différent.
   * Cela redirige l'utilisateur vers un écran où il peut choisir une méthode AMF alternative (p. ex., application OTP, code de récupération).
   *
   * @param {PickAuthenticatorOptions} [payload] - Options personnalisées facultatives à inclure avec la requête.
   * @returns {Promise<void>} Une promesse qui se résout lorsque l'action est soumise.
   * @throws {Error} Génère une erreur si la soumission échoue.
   * @example
   * ```typescript
   * import MfaPhoneChallenge from '@auth0/auth0-acul-js/mfa-phone-challenge';
   *
   * const mfaPhoneChallenge = new MfaPhoneChallenge();
   * await mfaPhoneChallenge.tryAnotherMethod();
   * ```
   */
  tryAnotherMethod(payload?: PickAuthenticatorOptions): Promise<void>;
}

<ParamField body="continue" type="Promise<void>">
  continue
  Soumet le choix de la méthode d’envoi de l’AMF (SMS ou appel vocal) pour passer à l’étape de vérification.
  Cela déclenche l’envoi du code de vérification au téléphone de l’utilisateur.

  Une Promise qui se résout une fois l’action soumise.

  #### Exceptions

  Lève une erreur si la soumission échoue.

  ```typescript Example
  import MfaPhoneChallenge from '@auth0/auth0-acul-js/mfa-phone-challenge';

  const mfaPhoneChallenge = new MfaPhoneChallenge();

  // Pour envoyer le code par SMS
  await mfaPhoneChallenge.continue({ type: 'sms' });

  // Pour envoyer le code par appel vocal
  await mfaPhoneChallenge.continue({ type: 'voice' });
  ```
