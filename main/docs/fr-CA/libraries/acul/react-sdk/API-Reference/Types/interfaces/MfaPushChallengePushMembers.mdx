```ts Example
export interface MfaPushChallengePushMembers extends BaseMembers {
  screen: ScreenMembersOnMfaPushChallengePush;
  untrustedData: UntrustedDataMembersOnMfaPushChallengePush;

  /**

   */

  /**
   * Renvoie la notification push
   * @param payload Options personnalisées facultatives à inclure avec la requête
   */

  /**
   * Bascule vers la saisie manuelle du code de vérification
   * @param payload Options personnalisées facultatives à inclure avec la requête
   */

  /**
   * Permet d'essayer une autre méthode d'authentification
   * @param payload Options personnalisées facultatives à inclure avec la requête
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;

  /** 
   * Gère le polling pour le statut du défi push MFA
   * @param options Options de configuration pour le polling de type {@link MfaPollingOptions}
   * @returns Un objet pour contrôler le processus de polling
  */

}
```
