title: modèle de scripts Delete
---

Le script Delete implémente la fonction définie pour supprimer l’utilisateur indiqué d’une base de données externe. Nous recommandons de nommer cette fonction `deleteUser`.

Ce script n’est utilisé que dans un [scénario d’authentification héritée](/docs/fr-CA/authenticate/database-connections/custom-db/overview-custom-db-connections), et il est requis si vous souhaitez supprimer un utilisateur d’Auth0 et de votre base de données externe en une seule opération.

<div id="deleteuser-function">
  ## Fonction DeleteUser
</div>

La fonction `deleteUser` doit :

* Supprimer l’utilisateur indiqué de la base de données externe.
* Retourner la valeur `null` si la suppression a réussi.
* Retourner une erreur si la suppression a échoué.

<div id="definition">
  ### Définition
</div>

La fonction `deleteUser` accepte deux paramètres et renvoie une fonction `callback` :

```js lines
deleteUser(id, callback): function
```

| Paramètre  | Type     | Description                                                                 |
| ---------- | -------- | --------------------------------------------------------------------------- |
| `id`       | String   | Le `user_id` de l’identité de l’utilisateur. Ne commence pas par `auth0`.   |
| `callback` | Fonction | Permet de transmettre une erreur ou des données de profil dans le pipeline. |

### Exemple

Voici un exemple en pseudo-JavaScript montrant comment vous pourriez implémenter la fonction `deleteUser`. Pour des exemple de scripts par langage, consultez [exemple de scripts par langage](/docs/fr-CA/authenticate/database-connections/custom-db/templates/delete#language-specific-script-examples).

```javascript lines
function deleteUser(id, callback) {
  // Envoyer l'identifiant utilisateur à l'API de base de données externe
  let options = {
    url: "https://example.com/api/deleteUser",
    body: {
      id: id
    }
  };

  send(options, (err, profileData) => {
    // Retourner une erreur dans le callback si la suppression a échoué
    if (err) {
      return callback(new Error("My custom error message."));
    }

    // Retourner une valeur nulle dans le callback si la suppression a réussi
    return callback(null);
  });
}
```

<div id="callback-function">
  ## Fonction callback
</div>

La fonction `callback` prend un paramètre et retourne une fonction.

<div id="definition">
  ### Définition
</div>

```js lines
callback(error): function
```

| Paramètre | Type  | Obligatoire | Description                    |
| --------- | ----- | ----------- | ------------------------------ |
| `error`   | Objet | Obligatoire | Contient des données d’erreur. |

<div id="return-a-success">
  ### Indiquer la réussite
</div>

Si l’utilisateur a bien été supprimé de votre base de données externe, appelez la fonction `callback` et transmettez une valeur `null` pour le paramètre `error`.

<div id="example">
  #### Exemple
</div>

```js lines
return callback(null);
```

<div id="return-an-error">
  ### Retourner une erreur
</div>

Si une erreur se produit, appelez la fonction `callback` et transmettez les renseignements pertinents sur l’erreur au paramètre `error`.

<div id="example">
  #### Exemple
</div>

```js lines
return callback(new Error("My custom error message."));
```

<div id="javascript">
  ### JavaScript
</div>

```javascript lines
function remove (id, callback) {
  // Ce script supprime un utilisateur de votre base de données existante.
  // Il est exécuté chaque fois qu'un utilisateur est supprimé via la Management API ou le Auth0 Dashboard.
  //
  // Ce script peut se terminer de deux façons :
  // 1. L'utilisateur a été supprimé avec succès :
  //     callback(null);
  // 2. Un problème est survenu lors de la tentative d'accès à votre base de données :
  //     callback(new Error("my error message"));
  var msg = "Please implement the Delete script for this database " +
       "connection at https://manage.auth0.com/#/connections/database";
  return callback(new Error(msg));
}
```

<div id="aspnet-membership-provider-mvc3-universal-providers">
  ### ASP.NET Membership Provider (MVC3 - Universal Providers)
</div>

```javascript lines expandable
function remove(id, callback) {
  const sqlserver = require('tedious@1.11.0');
  const Connection = sqlserver.Connection;
  const Request = sqlserver.Request;
  const TYPES = sqlserver.TYPES;
  const connection = new Connection({
    userName: 'the username',
    password: 'the password',
    server: 'the server',
    options: {
      database: 'the db name',
      encrypt: true,
      // Requis pour récupérer l'userId nécessaire à la création de l'entité Membership
      rowCollectionOnRequestCompletion: true
    }
  });
  connection.on('debug', function(text) {
    // si vous avez des problèmes de connexion, décommentez ceci pour obtenir des informations plus détaillées
    // console.log(text);
  }).on('errorMessage', function(text) {
    // ceci affichera toutes les erreurs lors de la connexion à la base de données SQL ou avec les instructions SQL
    console.log(JSON.stringify(text));
  });
  connection.on('connect', function(err) {
    if (err) return callback(err);
    executeDelete(['Memberships', 'Users'], function(err) {
      if (err) return callback(err);
      callback(null);
    });
  });
  function executeDelete(tables, callback) {
    const query = tables.map(function(table) {
      return 'DELETE FROM ' + table + ' WHERE UserId = @UserId';
    }).join(';');
    const request = new Request(query, function(err) {
      if (err) return callback(err);
      callback(null);
    });
    request.addParameter('UserId', TYPES.VarChar, id);
    connection.execSql(request);
  }
}
```

<div id="mongodb">
  ### MongoDB
</div>

```javascript lines
function remove(id, callback) {
  const MongoClient = require('mongodb@3.1.4').MongoClient;
  const client = new MongoClient('mongodb://user:pass@mymongoserver.com');
  client.connect(function (err) {
    if (err) return callback(err);
    const db = client.db('db-name');
    const users = db.collection('users');
    users.remove({ _id: id }, function (err) {
      client.close();
      if (err) return callback(err);
      callback(null);
    });
  });
}
```

<div id="postgresql">
  ### PostgreSQL
</div>

```javascript lines
function remove(id, callback) {
  //cet exemple utilise la bibliothèque "pg"
  //plus d'infos ici : https://github.com/brianc/node-postgres
  const postgres = require('pg');
  const conString = 'postgres://user:pass@localhost/mydb';
  postgres.connect(conString, function (err, client, done) {
    if (err) return callback(err);
    const query = 'DELETE FROM users WHERE id = $1';
    client.query(query, [id], function (err) {
      // REMARQUE : appelez toujours `done()` ici pour fermer
      // la connexion à la base de données
      done();
      return callback(err);
    });
  });
}
```

  // plus d'infos ici : http://pekim.github.io/tedious/index.html
