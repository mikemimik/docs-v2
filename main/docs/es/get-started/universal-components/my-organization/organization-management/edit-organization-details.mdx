      ```tsx
      import React from "react";
      import { OrganizationDetailsEdit } from "@auth0/universal-components-react";
      import { Auth0Provider } from "@auth0/auth0-react";
      import { Auth0ComponentProvider } from "@auth0/universal-components-react/spa";
      import { useNavigate } from "react-router-dom";
      import { analytics } from "./lib/analytics";

      function OrganizationSettingsPage() {
        const navigate = useNavigate();

        const handleSaveSuccess = (org) => {
          analytics.track("Organization Updated", {
            name: org.name,
            displayName: org.display_name,

          navigate("/organization");

                  return confirm(`Save changes to "${org.display_name}"?`);
                },
                onAfter: handleSaveSuccess,

                classes: {
                  "OrganizationDetailsEdit-form": "shadow-xl rounded-lg",
                },
              }}
            />

            </Auth0ComponentProvider>

        );
      }
      ```

    **saveAction**

    **Type:** `ComponentAction<OrganizationPrivate>`

    Controla el flujo de guardado cuando los usuarios envían cambios en la organización. Usa `onAfter` para navegar a otra página después de guardar correctamente.

    **Propiedades:**

    * `disabled` - Desactiva el botón Guardar (p. ej., mientras otra operación está en curso)
    * `onBefore(data)` - Se ejecuta antes de la operación de guardado. Devuelve `false` para impedir que se guarde (p. ej., para mostrar primero un cuadro de diálogo de confirmación).
    * `onAfter(data)` - Se ejecuta después de que la organización se haya guardado correctamente. Úsalo para ir a otra página o registrar el evento.

    **Ejemplo:**

    ```tsx
    // Volver atrás después de guardar
    saveAction={{
      onAfter: () => navigate("/organization"),
    }}

    // Agregar confirmación antes de guardar
    saveAction={{
      onBefore: (org) => {
        return confirm(`Save changes to "${org.display_name}"?`);
      },
      onAfter: () => navigate("/organization"),
    }}

    // Registrar analíticas al guardar
    saveAction={{
      onAfter: (org) => {
        analytics.track("Organization Updated", {
          name: org.name,
          displayName: org.display_name,
        });
        navigate("/organization");
      },
    }}
    ```

    ***

    **cancelAction**

    **Type:** `ComponentAction<OrganizationPrivate>`

    Controla el flujo de cancelación/descarte. Usa esta Action para gestionar los cambios descartados.

    **Propiedades:**

    * `disabled` - Desactiva el botón Cancelar
    * `onBefore(data)` - Se ejecuta antes de la acción de cancelación. Devuelve `false` para impedir la cancelación (p. ej., para confirmar que se descarten los cambios no guardados).
    * `onAfter(data)` - Se ejecuta después de confirmar la cancelación. Úsalo para salir del formulario.

    **Ejemplo:**

    ```tsx
    <OrganizationDetailsEdit
      cancelAction={{
        onBefore: (org) => {
          return confirm("Discard unsaved changes?");
        },
        onAfter: () => navigate("/organization"),
      }}
    />
    ```

    ***

    **backButton**

    **Type:** `{ icon?: LucideIcon; onClick: (e: MouseEvent) => void }`

    Configura el botón de retroceso en el encabezado del componente. Úsalo para volver a la información general de tu organización o a la página anterior.

    **Propiedades:**

    * `icon` - componente de icono Lucide personalizado (opcional; el valor predeterminado es ArrowLeft)
    * `onClick` - controlador de clic para la navegación

    **Ejemplo:**

    ```tsx
    import { ChevronLeft } from "lucide-react";

    <OrganizationDetailsEdit
      backButton={{
        icon: ChevronLeft,
        onClick: () => navigate("/organization"),
      }}
    />;
    ```

    **schema**

    Establece reglas de validación personalizadas para los campos de detalles de la organización.

    <Accordion title="Campos disponibles del esquema">
      Todos los campos del esquema admiten: `regex`, `errorMessage`, `minLength`, `maxLength`, `required`

      **details.name** - Nombre interno de la organización
      **details.displayName** - Nombre para mostrar de la organización
      **details.color** - Color de la marca (formato hexadecimal)
      **details.logoURL** - URL del logotipo
    </Accordion>

    ```tsx
    <OrganizationDetailsEdit
      schema={{
        details: {
          name: {
            minLength: 3,
            maxLength: 50,
            regex: /^[a-zA-Z0-9-_]+$/,
            errorMessage: "Name must be alphanumeric with hyphens/underscores",
          },
          displayName: {
            required: true,
            maxLength: 100,
          },
          color: {
            regex: /^#[0-9A-Fa-f]{6}$/,
            errorMessage: "Enter a valid hex color (e.g., #FF5500)",
          },
          logoURL: {
            regex: /^https:\/\/.+/,
            errorMessage: "Logo URL must use HTTPS",
          },
        },
      }}
    />
    ```

    **saveAction**

    **Type:** `ComponentAction<OrganizationPrivate>`

    Controla el flujo de guardado cuando los usuarios envían cambios en la organización. Usa `onAfter` para navegar a otra página después de guardar correctamente.

    **Propiedades:**

    * `disabled` - Deshabilita el botón Guardar (p. ej., mientras hay otra operación en curso)
    * `onBefore(data)` - Se ejecuta antes de la operación de guardado. Devuelve `false` para impedir que se guarde (p. ej., para mostrar antes un cuadro de diálogo de confirmación).
    * `onAfter(data)` - Se ejecuta después de que la organización se haya guardado correctamente. Úsalo para ir a otra página o registrar el evento.

    **Patrones comunes:**

    ```tsx
    // Volver atrás después de guardar
    saveAction={{
      onAfter: () => navigate("/organization"),
    }}

    // Agregar confirmación antes de guardar
    saveAction={{
      onBefore: (org) => {
        return confirm(`Save changes to "${org.display_name}"?`);
      },
      onAfter: () => navigate("/organization"),
    }}

    // Registrar analíticas al guardar
    saveAction={{
      onAfter: (org) => {
        analytics.track("Organization Updated", {
          name: org.name,
          displayName: org.display_name,
        });
        navigate("/organization");
      },
    }}
    ```

    ***

    **cancelAction**

    **Type:** `ComponentAction<OrganizationPrivate>`

    Controla el flujo de cancelación/descarte. Usa esta Action para gestionar los cambios descartados.

    **Propiedades:**

    * `disabled` - Desactiva el botón Cancelar
    * `onBefore(data)` - Se ejecuta antes de la acción de cancelación. Devuelve `false` para evitar la cancelación (p. ej., para confirmar que quieres descartar los cambios no guardados).
    * `onAfter(data)` - Se ejecuta después de confirmar la cancelación. Úsalo para salir del formulario.

    **Ejemplo:**

    ```tsx
    <OrganizationDetailsEdit
      cancelAction={{
        onBefore: (org) => {
          return confirm("Discard unsaved changes?");
    ```
