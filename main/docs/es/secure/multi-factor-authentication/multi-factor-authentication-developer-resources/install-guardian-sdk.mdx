El SDK de Guardian proporciona un cliente sin interfaz de usuario para Guardian.

`npm install auth0-guardian-js`

<div id="configure-guardian">
  ## Configurar Guardian
</div>

```javascript lines
var auth0GuardianJS = require('auth0-guardian-js')({
	// Para tenants de EE. UU.: https://{name}.guardian.auth0.com
 	// Para tenants de AU: https://{name}.guardian.au.auth0.com
 	// Para tenants de EU: https://{name}.guardian.eu.auth0.com
  // Para tenants de JP: https://{name}.guardian.jp.auth0.com
	serviceUrl: "https://{{ userData.tenant }}.guardian.auth0.com",
	requestToken: "{{ requestToken }}", // o ticket: "{{ ticket }}" - ver más abajo

	issuer: {
		// El nombre del emisor que se mostrará en las aplicaciones de generación de OTP
		label: "{{ userData.tenantFriendlyName }}",
		name: "{{ userData.tenant }}",
	},

	// El label de la cuenta que se mostrará en las aplicaciones de generación de OTP
	accountLabel: "{{ userData.friendlyUserId }}",

	// Opcional, solo para fines de depuración,
	// ID que permite asociar un grupo de requests
	// como pertenecientes a la misma "transaction" (en sentido amplio)
	globalTrackingId: "{{ globalTrackingId }}"
});
```

El uso de `requestToken` o `ticket` depende del método de autenticación. Ticket corresponde a un [ticket de inscripción](/docs/es/secure/multi-factor-authentication/multi-factor-authentication-developer-resources/create-custom-enrollment-tickets) generado previamente.

<div id="use-a-custom-domain">
  ### Usar un dominio personalizado
</div>

Si tienes un <Tooltip tip="Custom Domain: Dominio de terceros con un nombre especializado o personalizado." cta="Ver glosario" href="/docs/es/glossary?term=custom+domain">dominio personalizado</Tooltip> configurado para tu tenant de Auth0, deberás actualizar la propiedad `serviceUrl` para que apunte al endpoint de Guardian:

```javascript lines
var auth0GuardianJS = require('auth0-guardian-js')({
    // Para dominios personalizados
    serviceUrl: "https://{yourCustomDomain}/guardian/",
    ...
});
```

<div id="enroll-devices">
  ## Registrar dispositivos
</div>

El registro de dispositivos consta de los siguientes pasos:

1. Inicia la transacción.
2. (opcional) Comprueba si el usuario ya está registrado. No puedes registrarlo dos veces.
3. Envía la información necesaria para registrarlo.
4. Confirma el registro.
5. Muestra el código de recuperación.

Algunos pasos pueden omitirse según el método. Ofrecemos la misma interfaz para todos los métodos para que puedas escribir un código uniforme. Algunos métodos completan la autenticación, mientras que otros requieren un paso adicional de autenticación. Puedes determinarlo escuchando el evento `enrollment-complete`.

```javascript lines expandable
function enroll(transaction, method) {
	if (transaction.isEnrolled()) {
		console.log('You are already enrolled');
		return;
	}

	var enrollData = {};

	if (method === 'sms') {
		enrollData.phoneNumber = prompt('Phone number'); // Recopilar número de teléfono
	}

	return transaction.enroll(method, enrollData, function (err, otpEnrollment) {
		if (err) {
			console.error(err);
			return;
		}

		var uri = otpEnrollment.getUri();
		if (uri) {
			showQR(uri);
		}

		var confirmData = {};
		if (method === 'otp' || method === 'sms') {
			confirmData.otpCode = prompt('Otp code'); // Collect verification otp
		}

		otpEnrollment.confirm(confirmData);
	});
}

auth0GuardianJS.start(function(err, transaction) {
	if (err) {
		console.error(err);
		return;
	}

	transaction.on('error', function(error) {
		console.error(error);
	});

	transaction.on('timeout', function() {
		console.log('Timeout');
	});

	transaction.on('enrollment-complete', function(payload) {
		if (payload.recoveryCode) {
			alert('Recovery code is ' + payload.recoveryCode);
		}

		if (payload.authRequired) {
			showAuthenticationFor(transaction, payload.enrollment);
			return;
		}
	});

	transaction.on('auth-response', function(payload) {
		if (payload.recoveryCode) {
			alert('The new recovery code is ' + payload.recoveryCode);
		}

		if (!payload.accepted) {
			alert('Authentication has been rejected');
			return;
		}

		auth0GuardianJS.formPostHelper('{{ postActionURL }}', { signature: payload.signature });
	});

	var availableEnrollmentMethods = transaction.getAvailableEnrollmentMethods();

	method = prompt('What method do you want to use, select one of '
		+ availableEnrollmentMethods.join(', '));

	enroll(transaction, method) // For sms
});
```

<div id="authenticate">
  ## Autenticarse
</div>

Para autenticarse con un método, debe ejecutar los siguientes pasos:

1. Inicie la transacción.
2. (opcional) Compruebe si el usuario ya está registrado. Debe estar registrado para autenticarse.
3. Solicite la autenticación (la notificación push o el SMS). La solicitud es un noop para OTP.
4. Verifique el OTP (`.verify` es un noop para push).

Algunos pasos pueden omitirse según el método; ofrecemos la misma interfaz para todos los métodos para que pueda escribir un código uniforme. Después de que se verifique el factor o se acepte la solicitud push, recibirá un evento `auth-response` con el payload para enviar al servidor. Puede usar `auth0GuardianJS.formPostHelper('{{ postActionURL }}', payload)` para devolver el mensaje al servidor.

También puede recibir `auth-rejected` si se recibió la notificación push.

```javascript lines expandable
function authenticate(method) {
	auth0GuardianJS.start(function (err, transaction) {
		if (err) {
			console.error(err);
			return;
		}

		if (!transaction.isEnrolled()) {
			console.log('You are not enrolled');
			return;
		}

		transaction.on('error', function(error) {
			console.error(error);
		});

		transaction.on('timeout', function() {
			console.log('Timeout');
		});

		transaction.on('auth-response', function(payload) {
			if (payload.recoveryCode) {
				alert('The new recovery code is ' + payload.recoveryCode);
			}

			if (!payload.accepted) {
				alert('Authentication has been rejected');
				return;
			}

			auth0GuardianJS.formPostHelper('{{ postActionURL }}', { signature: payload.signature });
		});

		var enrollment = transaction.getEnrollments()[0];

		if (enrollment.getAvailableAuthenticatorTypes().length === 0) {
			alert('Somethings went wrong, seems that there is no authenticators');
			return;
		}

		transaction.requestAuth(enrollment, { method: method } function(err, auth) {
			if (err) {
				console.error(err);
				return;
			}

			var data = {};
			if (method === 'sms' || method === 'otp') {
				data.otpCode = prompt('Otp code');
			} else if (method === 'recovery-code') {
				data.recoveryCode = prompt('Recovery code');
			}

			return auth.verify(data);
		});
	});
}
```
