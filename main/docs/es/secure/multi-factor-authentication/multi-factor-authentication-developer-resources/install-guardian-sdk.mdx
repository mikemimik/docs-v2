El SDK de Guardian proporciona un cliente sin interfaz gráfica para Guardian.

`npm install auth0-guardian-js`

<div id="configure-guardian">
  ## Configurar Guardian
</div>

```javascript lines
var auth0GuardianJS = require('auth0-guardian-js')({
	// Para tenants de EE. UU.: https://{name}.guardian.auth0.com
 	// Para tenants de AU: https://{name}.guardian.au.auth0.com
 	// Para tenants de EU: https://{name}.guardian.eu.auth0.com
  // Para tenants de JP: https://{name}.guardian.jp.auth0.com
	serviceUrl: "https://{{ userData.tenant }}.guardian.auth0.com",
	requestToken: "{{ requestToken }}", // o ticket: "{{ ticket }}" - ver más abajo

	issuer: {
		// El nombre del emisor que se mostrará en las apps de generación de OTP
		label: "{{ userData.tenantFriendlyName }}",
		name: "{{ userData.tenant }}",
	},

	// La etiqueta de cuenta que se mostrará en las apps de generación de OTP
	accountLabel: "{{ userData.friendlyUserId }}",

	// Opcional, solo para fines de depuración,
	// ID que permite asociar un grupo de requests
	// como pertenecientes a la misma "transaction" (en sentido amplio)
	globalTrackingId: "{{ globalTrackingId }}"
});
```

El uso de `requestToken` o `ticket` depende del método de autenticación. El `ticket` corresponde a un [ticket de inscripción generado previamente](/docs/es/secure/multi-factor-authentication/multi-factor-authentication-developer-resources/create-custom-enrollment-tickets).

<div id="use-a-custom-domain">
  ### Usa un dominio personalizado
</div>

Si tienes configurado un <Tooltip tip="Custom Domain: Dominio de terceros con un nombre especializado o personalizado." cta="Ver glosario" href="/docs/es/glossary?term=custom+domain">dominio personalizado</Tooltip> para tu tenant de Auth0, tendrás que actualizar la propiedad `serviceUrl` para que apunte al endpoint de Guardian:

```javascript lines
var auth0GuardianJS = require('auth0-guardian-js')({
    // Para dominios personalizados
    serviceUrl: "https://{yourCustomDomain}/guardian/",
    ...
});
```

<div id="enroll-devices">
  ## Inscribir dispositivos
</div>

La inscripción de dispositivos consta de los siguientes pasos:

1. Inicie la transacción.
2. (opcional) Compruebe si el usuario ya está inscrito. No puede inscribirse dos veces.
3. Envíe la información necesaria para inscribir el dispositivo.
4. Confirme la inscripción.
5. Muestre el código de recuperación.

Algunos pasos pueden omitirse según el método. Ofrecemos la misma interfaz para todos los métodos, de modo que pueda escribir un código uniforme. Algunos métodos completan la autenticación, mientras que otros requieren un paso de autenticación adicional. Puede determinarlo escuchando el evento `enrollment-complete`.

```javascript lines expandable
function enroll(transaction, method) {
	if (transaction.isEnrolled()) {
		console.log('You are already enrolled');
		return;
	}

	var enrollData = {};

	if (method === 'sms') {
		enrollData.phoneNumber = prompt('Phone number'); // Obtener número de teléfono
	}

	return transaction.enroll(method, enrollData, function (err, otpEnrollment) {
		if (err) {
			console.error(err);
			return;
		}

		var uri = otpEnrollment.getUri();
		if (uri) {
			showQR(uri);
		}

		var confirmData = {};
		if (method === 'otp' || method === 'sms') {
			confirmData.otpCode = prompt('Otp code'); // Collect verification otp
		}

		otpEnrollment.confirm(confirmData);
	});
}

auth0GuardianJS.start(function(err, transaction) {
	if (err) {
		console.error(err);
		return;
	}

	transaction.on('error', function(error) {
		console.error(error);
	});

	transaction.on('timeout', function() {
		console.log('Timeout');
	});

	transaction.on('enrollment-complete', function(payload) {
		if (payload.recoveryCode) {
			alert('Recovery code is ' + payload.recoveryCode);
		}

		if (payload.authRequired) {
			showAuthenticationFor(transaction, payload.enrollment);
			return;
		}
	});

	transaction.on('auth-response', function(payload) {
		if (payload.recoveryCode) {
			alert('The new recovery code is ' + payload.recoveryCode);
		}

		if (!payload.accepted) {
			alert('Authentication has been rejected');
			return;
		}

		auth0GuardianJS.formPostHelper('{{ postActionURL }}', { signature: payload.signature });
	});

	var availableEnrollmentMethods = transaction.getAvailableEnrollmentMethods();

	method = prompt('What method do you want to use, select one of '
		+ availableEnrollmentMethods.join(', '));

	enroll(transaction, method) // For sms
});
```

<div id="authenticate">
  ## Autenticar
</div>

Para autenticarte mediante un método, debes seguir estos pasos:

1. Inicia la transacción.
2. (opcional) Comprueba si el usuario ya está registrado. Debes estar registrado para autenticarte.
3. Solicita la autenticación (la notificación push o el SMS). En OTP, la solicitud no realiza ninguna acción.
4. Verifica el OTP (`.verify` no realiza ninguna acción para push).

Según el método, algunos pasos pueden omitirse; proporcionamos la misma interfaz para todos los métodos para que puedas escribir código de forma uniforme. Una vez que se verifique el factor o se acepte la notificación push, recibirás un evento `auth-response` con el payload que debes enviar al servidor. Puedes usar `auth0GuardianJS.formPostHelper('{{ postActionURL }}', payload)` para enviar el mensaje de vuelta al servidor.

También puedes recibir `auth-rejected` si se recibió la notificación push.

```javascript lines expandable
function authenticate(method) {
	auth0GuardianJS.start(function (err, transaction) {
		if (err) {
			console.error(err);
			return;
		}

		if (!transaction.isEnrolled()) {
			console.log('You are not enrolled');
			return;
		}

		transaction.on('error', function(error) {
			console.error(error);
		});

		transaction.on('timeout', function() {
			console.log('Timeout');
		});

		transaction.on('auth-response', function(payload) {
			if (payload.recoveryCode) {
				alert('The new recovery code is ' + payload.recoveryCode);
			}

			if (!payload.accepted) {
				alert('Authentication has been rejected');
				return;
			}

			auth0GuardianJS.formPostHelper('{{ postActionURL }}', { signature: payload.signature });
		});

		var enrollment = transaction.getEnrollments()[0];

		if (enrollment.getAvailableAuthenticatorTypes().length === 0) {
			alert('Somethings went wrong, seems that there is no authenticators');
			return;
		}

		transaction.requestAuth(enrollment, { method: method } function(err, auth) {
			if (err) {
				console.error(err);
				return;
			}

			var data = {};
			if (method === 'sms' || method === 'otp') {
				data.otpCode = prompt('Otp code');
			} else if (method === 'recovery-code') {
				data.recoveryCode = prompt('Recovery code');
			}

			return auth.verify(data);
		});
	});
}
```
