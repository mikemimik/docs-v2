---
description: Cómo migrar usuarios desde Azure a Auth0
title: Migrar usuarios desde Azure
---

Auth0 admite la importación de usuarios desde Azure mediante el [flujo ROPC de Azure AD](https://docs.microsoft.com/en-us/azure/active-directory/develop/v2-oauth-ropc) en los [scripts de inicio de sesión de base de datos personalizada](/docs/es/authenticate/database-connections/custom-db/templates/login) de Auth0.

1. [Cree un registro de aplicación](https://docs.microsoft.com/en-us/azure/active-directory/develop/quickstart-register-app) como **cliente público** en los tenants o directorios correspondientes de su aplicación y, a continuación, almacene de forma segura los ID de cliente y los secretos de cliente proporcionados para las llamadas a la API.
2. Cuando se le pregunte quién puede usar esta aplicación o acceder a esta API, elija **Cuentas en cualquier directorio organizativo**.
3. En la sección Microsoft Graph, conceda permisos a `Directory.Read.All` y `User.Read`.
4. Si su aplicación es B2C, [configure un flujo de credenciales de contraseña del propietario del recurso en Azure Active Directory B2C](https://learn.microsoft.com/en-us/azure/active-directory-b2c/add-ropc-policy?tabs=app-reg-ga\&pivots=b2c-user-flow).

Ahora ya puede usar el flujo ROPC. Los siguientes fragmentos muestran ejemplos de llamadas.

<div id="azure-ad-login-script-template">
  ##### Plantilla de script de inicio de sesión de Azure AD:
</div>

```javascript lines expandable
function login(email, password, callback) {
      const axios = require('axios').default;
      const qs = require('qs');
    	const jwtDecode = require('jwt-decode').default;

       const data = { 'client_id': {{YOUR_AZURE_APP_CLIENT_ID}},
             'scope': 'user.read openid profile offline_access',
             'username': email,
             'password': password,
             'grant_type': 'password',
             'client_secret': {{YOUR_AZURE_APP_CLIENT_SECRET}}
             };
const options = {
  method: 'POST',
  headers: { 'content-type': 'application/x-www-form-urlencoded' },
  data: qs.stringify(data),
  url: 'https://login.microsoftonline.com/{{AZURE_TENANT_ID}}/oauth2/v2.0/token',
};

axios(options).then(
	response => {
  // en un entorno de producción, lo ideal sería validar este JWT completamente antes de leer los claims
  const claims = jwtDecode(response.data.id_token);

  if (response.statusCode === 401) return callback();

  callback(null, {
  // este es un ejemplo sencillo de propiedades que pueden mapearse al perfil de usuario de auth0
  // puedes elegir exactamente qué se mapea según los datos que obtengas del token de azure
  user_id: claims.sub,
  nickname: claims.name,
  email: claims.preferred_username
});

}
      );

    }
```

<div id="b2c-login-script-template">
  ##### Plantilla del script de inicio de sesión B2C:
</div>

```javascript lines expandable
function login(email, password, callback) {
      const axios = require('axios').default;
      const qs = require('qs');
    	const jwtDecode = require('jwt-decode').default;

       const data = { 'client_id': {{YOUR_B2C_APP_CLIENT_ID}},
             'scope': 'openid {{YOUR_B2C_APP_CLIENT_ID}} offline_access profile',
             'username': email,
             'password': password,
             'grant_type': 'password',
             };
const options = {
  method: 'POST',
  headers: { 'content-type': 'application/x-www-form-urlencoded' },
  data: qs.stringify(data),
  url: 'https://{{TENANT_NAME}}.b2clogin.com/{{TENANT_NAME}}.onmicrosoft.com/{{B2C_POLICY_NAME}}/oauth2/v2.0/token',
};

axios(options).then(
	response => {
  // en un entorno de producción, lo ideal sería validar este JWT completamente antes de leer los claims
  const claims = jwtDecode(response.data.access_token);
  if (response.statusCode === 401) return callback();

  callback(null, {
  // este es un ejemplo sencillo de propiedades que pueden mapearse al perfil de usuario de auth0
  // puedes elegir exactamente qué incluir según los datos que obtengas del token de azure
  user_id: claims.sub,
  nickname: claims.given_name,
  email: claims.emails[0]
});
}
      );
    }
```
