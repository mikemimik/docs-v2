---
title: "MfaVoiceChallengeMembers"
---

Interfaz que define todos los miembros y las operaciones disponibles en la pantalla del desafío de voz de MFA.

````ts Example
export interface MfaVoiceChallengeMembers extends BaseMembers {
  /**
   * Propiedades y datos específicos de la pantalla.
   */
  screen: ScreenMembersOnMfaVoiceChallenge;

  untrustedData: UntrustedDataMembersOnMfaVoiceChallenge;

  /**
   * Envía el código de verificación por voz para validar el desafío MFA.
   *
   * @param payload - Objeto que contiene el código de verificación y los parámetros opcionales
   * @returns Promise que se resuelve cuando el código se valida correctamente
   *
   * @example
   * ```typescript
   * const mfaVoiceChallenge = new MfaVoiceChallenge();
   * mfaVoiceChallenge.continue({
   *   code: '123456',
   *   rememberDevice: true
   * });
   * ```
   */
  continue(payload: MfaVoiceChallengeContinueOptions): Promise<void>;

  /**
   * Navega a la pantalla para seleccionar un número de teléfono diferente.
   *
   * @param payload - Parámetros personalizados opcionales
   * @returns Promise que se resuelve cuando la navegación finaliza
   *
   * @example
   * ```typescript
   * const mfaVoiceChallenge = new MfaVoiceChallenge();
   * mfaVoiceChallenge.pickPhone();
   * ```
   */
  pickPhone(payload?: CustomOptions): Promise<void>;

  /**
   * Cambia al método de verificación por SMS en lugar de la llamada de voz.
   *
   * @param payload - Parámetros personalizados opcionales
   * @returns Promise que se resuelve cuando el cambio finaliza
   *
   * @example
   * ```typescript
   * const mfaVoiceChallenge = new MfaVoiceChallenge();
   * mfaVoiceChallenge.switchToSms();
   * ```
   */
  switchToSms(payload?: CustomOptions): Promise<void>;

  /**
   * Solicita una nueva llamada de voz con un código de verificación.
   *
   * @param payload - Parámetros personalizados opcionales
   * @returns Promise que se resuelve cuando se envía el nuevo código
   *
   * @example
   * ```typescript
   * const mfaVoiceChallenge = new MfaVoiceChallenge();
   * mfaVoiceChallenge.resendCode();
   * ```
   */
  resendCode(payload?: CustomOptions): Promise<void>;

  /**
   * Navega a la pantalla para seleccionar un método MFA alternativo.
   *
   * @param payload - Parámetros personalizados opcionales
   * @returns Promise que se resuelve cuando la navegación finaliza
   *
   * @example
   * ```typescript
   * const mfaVoiceChallenge = new MfaVoiceChallenge();
   * mfaVoiceChallenge.tryAnotherMethod();
   * ```
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;

  /**
   * Obtiene la funcionalidad de reenvío con gestión del tiempo de espera para esta pantalla
   * @param options Opciones de configuración para la funcionalidad de reenvío
   */
  resendManager(options?: StartResendOptions): ResendControl;
}
````
