MfaPhoneChallengeMembers

````ts Example
export interface MfaPhoneChallengeMembers extends BaseMembers {
  /**
   * @property {ScreenMembersOnMfaPhoneChallenge} screen - Información específica de la pantalla para el MFA Phone Challenge.
   */
  screen: ScreenMembersOnMfaPhoneChallenge;

  /**
   * continue
   * Envía la elección del método de entrega de MFA (SMS o voz) para continuar con el desafío.
   * Esto activará el envío del código de verificación al teléfono del usuario.
   *
   * @param {ContinueOptions} payload - Las opciones que especifican el tipo de entrega ('sms' o 'voice').
   * @returns {Promise<void>} Una promesa que se resuelve cuando se envía la acción.
   * @throws {Error} Lanza un error si el envío falla.
   * @example
   * ```typescript
   * import MfaPhoneChallenge from '@auth0/auth0-acul-js/mfa-phone-challenge';
   *
   * const mfaPhoneChallenge = new MfaPhoneChallenge();
   *
   * // Para enviar el código por SMS
   * await mfaPhoneChallenge.continue({ type: 'sms' });
   *
   * // Para enviar el código por llamada de voz
   * await mfaPhoneChallenge.continue({ type: 'voice' });
   * ```
   */
  continue(payload: ContinueOptions): Promise<void>;

  /**
   * pickPhone
   * Inicia la acción para seleccionar un número de teléfono diferente.
   * Por lo general, redirige al usuario a una pantalla donde puede elegir entre varios números de teléfono registrados, si corresponde.
   *
   * @param {PickPhoneOptions} [payload] - Opciones personalizadas opcionales para incluir en la solicitud.
   * @returns {Promise<void>} Una promesa que se resuelve cuando se envía la acción.
   * @throws {Error} Lanza un error si el envío falla.
   * @example
   * ```typescript
   * import MfaPhoneChallenge from '@auth0/auth0-acul-js/mfa-phone-challenge';
   *
   * const mfaPhoneChallenge = new MfaPhoneChallenge();
   * await mfaPhoneChallenge.pickPhone();
   * ```
   */
  pickPhone(payload?: PickPhoneOptions): Promise<void>;

  /**
   * tryAnotherMethod
   * Inicia la acción para seleccionar un factor/autenticador de MFA diferente.
   * Redirige al usuario a una pantalla donde puede elegir un método de MFA alternativo (por ejemplo, una app OTP o un código de recuperación).
   *
   * @param {PickAuthenticatorOptions} [payload] - Opciones personalizadas opcionales para incluir en la solicitud.
   * @returns {Promise<void>} Una promesa que se resuelve cuando se envía la acción.
   * @throws {Error} Lanza un error si el envío falla.
   * @example
   * ```typescript
   * import MfaPhoneChallenge from '@auth0/auth0-acul-js/mfa-phone-challenge';
   *
   * const mfaPhoneChallenge = new MfaPhoneChallenge();
   * await mfaPhoneChallenge.tryAnotherMethod();
   * ```
   */
  tryAnotherMethod(payload?: PickAuthenticatorOptions): Promise<void>;
}

<ParamField body="continue" type="Promise<void>">
  continue
  Envía la selección del método de entrega de MFA (SMS o voz) para continuar con el desafío.
  Esto enviará el código de verificación al teléfono del usuario.

  Una promesa que se resuelve cuando se envía la acción.

  #### Lanza

  Lanza un error si falla el envío.

  ```typescript Example
  import MfaPhoneChallenge from '@auth0/auth0-acul-js/mfa-phone-challenge';

  const mfaPhoneChallenge = new MfaPhoneChallenge();

  // Para enviar el código por SMS
  await mfaPhoneChallenge.continue({ type: 'sms' });

  // Para enviar el código mediante llamada de voz
  await mfaPhoneChallenge.continue({ type: 'voice' });
  ```
