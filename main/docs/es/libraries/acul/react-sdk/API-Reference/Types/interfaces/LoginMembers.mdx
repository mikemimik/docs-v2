---
title: "LoginMembers"
---

Interfaz de miembros de la pantalla de inicio de sesión que amplía los miembros básicos

```ts Example
export interface LoginMembers extends BaseMembers {
  screen: ScreenMembersOnLogin;
  transaction: TransactionMembersOnLogin;
  /**
   * Realiza el inicio de sesión con nombre de usuario/contraseña
   * @param payload Las opciones de inicio de sesión
   */
  login(payload: LoginOptions): Promise<void>;
  /**
   * Realiza el inicio de sesión con proveedor social
   * @param payload Las opciones de inicio de sesión social
   */
  federatedLogin(payload: FederatedLoginOptions): Promise<void>;
  /**
   * Selecciona el código de país para la entrada del número de teléfono
   * @param payload Opciones Custom opcionales
   */
  pickCountryCode(payload?: CustomOptions): Promise<void>;
  /**
   * Obtiene los tipos de identificadores activos para la pantalla de inicio de sesión
   * @returns Un arreglo de tipos de identificadores activos o null si ninguno está activo
   * @utilityFeature
   */
  getLoginIdentifiers(): IdentifierType[] | null;
}
```
