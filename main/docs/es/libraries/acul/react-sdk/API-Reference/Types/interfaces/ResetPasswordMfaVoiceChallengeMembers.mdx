Interfaz que define los métodos y propiedades disponibles de la pantalla reset-password-mfa-voice-challenge.

````ts Example
export interface ResetPasswordMfaVoiceChallengeMembers extends BaseMembers {
  screen: ScreenMembersOnResetPasswordMfaVoiceChallenge;

  /**
   * Continúa con el desafío de voz usando el código proporcionado.
   *
   * @param payload - Las opciones que contienen el código.
   * @returns Una promesa que se resuelve cuando el desafío es enviado.
   *
   * @example
   * ```ts
   * const reset = new ResetPasswordMfaVoiceChallenge();
   * await reset.continue({ code: '123456' });
   * ```
   */
  continue(payload: ContinueOptions): Promise<void>;

  /**
   * Cambia a la verificación por SMS.
   *
   * @param payload - Opciones personalizadas opcionales para incluir con la solicitud.
   * @returns Una promesa que se resuelve cuando la acción se completa.
   */
  switchToSms(payload?: CustomOptions): Promise<void>;

  /**
   * Reenvía el código mediante llamada de voz.
   *
   * @param payload - Opciones personalizadas opcionales para incluir con la solicitud.
   * @returns Una promesa que se resuelve cuando el código es reenviado.
   */
  resendCode(payload?: CustomOptions): Promise<void>;

  /**
   * Permite al usuario intentar otro método de MFA.
   *
   * @param payload - Opciones personalizadas opcionales para incluir con la solicitud.
   * @returns Una promesa que se resuelve cuando la solicitud es enviada.
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;

  /**
   * Obtiene la funcionalidad de reenvío con gestión del tiempo de espera para esta pantalla
   * @param options Opciones de configuración para la funcionalidad de reenvío
   */
  resendManager(options?: StartResendOptions): ResendControl;
}
````
