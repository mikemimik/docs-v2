MfaWebAuthnRoamingChallengeMembers

````ts Example
export interface MfaWebAuthnRoamingChallengeMembers extends BaseMembers {
  /**
   * Propiedades y datos específicos de la pantalla, incluidas las opciones del desafío de WebAuthn.
   * @type {ScreenMembersOnMfaWebAuthnRoamingChallenge}
   */
  screen: ScreenMembersOnMfaWebAuthnRoamingChallenge;

  /**
   * Inicia el desafío de WebAuthn llamando a `navigator.credentials.get()` con las
   * opciones proporcionadas en `screen.data.publicKeyChallengeOptions`.
   * Si tiene éxito, envía la credencial resultante a Auth0 con `action: "default"`.
   * Si `navigator.credentials.get()` falla con un error conocido de la API de WebAuthn (como `NotAllowedError`),
   * este método llamará internamente a `reportWebAuthnError` para notificar a Auth0.
   *
   * @param {VerifySecurityKeyOptions} [options] - Parámetros opcionales para el proceso de verificación,
   * como `rememberDevice` y otras opciones personalizadas.
   * @returns {Promise<void>} Una promesa que se resuelve cuando el intento de verificación es enviado.
   *                          No devuelve datos directamente, ya que una operación exitosa generalmente resulta en una redirección.
   * @throws {Error} Lanza un error si `screen.data.publicKeyChallengeOptions` no está disponible,
   *                 si `navigator.credentials.get()` falla con un error inesperado,
   *                 o si el envío del formulario a Auth0 falla.
   *
   * @example
   * ```typescript
   * // Suponiendo que 'sdk' es una instancia de MfaWebAuthnRoamingChallenge
   * try {
   *   await sdk.verify({ rememberDevice: true });
   *   // Si tiene éxito, Auth0 gestiona la redirección.
   * } catch (error) {
   *   console.error("Security key verification failed:", error);
   *   // Comprueba sdk.transaction.errors para mensajes de validación del servidor si la página se recarga.
   * }
   * ```
   */
  verify(options?: VerifySecurityKeyOptions): Promise<void>;

  /**
   * Notifica a Auth0 un error específico de la API de WebAuthn.
   * Este método debe usarse si `navigator.credentials.get()` falla de una manera que
   * `verify()` no gestiona automáticamente, o si el desarrollador desea notificar
   * explícitamente un error antes de intentar otra acción.
   * Envía los detalles del error con `action: "showError::{errorDetails}"`.
   *
   * @param {ReportWebAuthnErrorOptions} options - Contiene el objeto `error` con `name` y `message`
   * de la API de WebAuthn, y cualquier otra opción personalizada.
   * @returns {Promise<void>} Una promesa que se resuelve cuando el informe de error es enviado.
   * @throws {Error} Lanza un error si el envío del formulario falla.
   *
   * @example
   * ```typescript
   * // Suponiendo que 'sdk' es una instancia de MfaWebAuthnRoamingChallenge
   * // Y webAuthnError es un objeto de error de una llamada fallida a navigator.credentials.get().
   * try {
   *   await sdk.reportWebAuthnError({
   *     error: { name: webAuthnError.name, message: webAuthnError.message }
   *   });
   * } catch (submitError) {
   *   console.error("Failed to report WebAuthn error:", submitError);
   * }
   * ```
   */
  reportWebAuthnError(options: ReportWebAuthnErrorOptions): Promise<void>;

  /**
   * Permite al usuario omitir el desafío de WebAuthn y seleccionar un método MFA diferente.
   * Esta acción envía `action: "pick-authenticator"` a Auth0, lo que debería dirigir
   * al usuario a la pantalla de selección de factor MFA.
   *
   * @param {TryAnotherMethodOptions} [options] - Parámetros personalizados opcionales para la solicitud.
   * @returns {Promise<void>} Una promesa que se resuelve cuando la acción es enviada.
   * @throws {Error} Lanza un error si el envío del formulario falla.
   *
   * @example
   * ```typescript
   * // Suponiendo que 'sdk' es una instancia de MfaWebAuthnRoamingChallenge
   * try {
   *   await sdk.tryAnotherMethod();
   *   // Si tiene éxito, Auth0 gestiona la redirección a la selección de MFA.
   * } catch (error) {
   *   console.error("Failed to switch MFA method:", error);
   * }
   * ```
   */
  tryAnotherMethod(options?: TryAnotherMethodOptions): Promise<void>;
}

<ParamField body="reportWebAuthnError" type="Promise<void>">
  Informa a Auth0 sobre un error específico de la API de WebAuthn.
  Este método debe usarse si `navigator.credentials.get()` falla de un modo que
  `verify()` no maneja automáticamente, o si el desarrollador quiere
  informar un error de forma explícita antes de intentar otra acción.
  Envía los detalles del error con `action: "showError::{errorDetails}"`.

  Una promesa que se resuelve cuando se envía el informe del error.

  #### Lanza

  Lanza un error si falla el envío del formulario.

  ```typescript Example
  // Suponiendo que 'sdk' es una instancia de MfaWebAuthnRoamingChallenge
  // Y webAuthnError es un objeto de error de una llamada fallida a navigator.credentials.get().

    await sdk.reportWebAuthnError({
      error: { name: webAuthnError.name, message: webAuthnError.message }
    });
  } catch (submitError) {
    console.error("Failed to report WebAuthn error:", submitError);
  }
  ```
