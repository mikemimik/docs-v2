MfaWebAuthnRoamingChallengeMembers

````ts Example
export interface MfaWebAuthnRoamingChallengeMembers extends BaseMembers {
  /**
   * Screen-specific properties and data, including WebAuthn challenge options.
   * @type {ScreenMembersOnMfaWebAuthnRoamingChallenge}
   */
  screen: ScreenMembersOnMfaWebAuthnRoamingChallenge;

  /**
   * Inicia el desafío de WebAuthn llamando a `navigator.credentials.get()` con las
   * opciones proporcionadas en `screen.data.publicKeyChallengeOptions`.
   * Si tiene éxito, envía la credencial resultante a Auth0 con `action: "default"`.
   * Si `navigator.credentials.get()` falla con un error conocido de la API de WebAuthn (como `NotAllowedError`),
   * este método llamará internamente a `reportWebAuthnError` para informar a Auth0.
   *
   * @param {VerifySecurityKeyOptions} [options] - Parámetros opcionales para el proceso de verificación,
   * como `rememberDevice` y otras opciones personalizadas.
   * @returns {Promise<void>} Una promesa que se resuelve cuando se envía el intento de verificación.
   *                          No devuelve datos directamente, ya que una operación exitosa generalmente resulta en una redirección.
   * @throws {Error} Lanza un error si falta `screen.data.publicKeyChallengeOptions`,
   *                 si `navigator.credentials.get()` falla con un error inesperado,
   *                 o si el envío del formulario a Auth0 falla.
   *
   * @example
   * ```typescript
   * // Assuming 'sdk' is an instance of MfaWebAuthnRoamingChallenge
   * try {
   *   await sdk.verify({ rememberDevice: true });
   *   // On success, Auth0 handles redirection.
   * } catch (error) {
   *   console.error("Security key verification failed:", error);
   *   // Check sdk.transaction.errors for server-side validation messages if the page reloads.
   * }
   * ```
   */
  verify(options?: VerifySecurityKeyOptions): Promise<void>;

  /**
   * Reports a specific API de WebAuthn error to Auth0.
   * This method should be used if `navigator.credentials.get()` fails in a way that
   * `verify()` doesn't automatically handle, or if the developer wants to explicitly
   * report an error before trying another action.
   * It submits the error details with `action: "showError::{errorDetails}"`.
   *
   * @param {ReportWebAuthnErrorOptions} options - Contains the `error` object with `name` and `message`
   * from the API de WebAuthn, and any other custom options.
   * @returns {Promise<void>} A promise that resolves when the error report is submitted.
   * @throws {Error} Throws an error if the form submission fails.
   *
   * @example
   * ```typescript
   * // Assuming 'sdk' is an instance of MfaWebAuthnRoamingChallenge
   * // And webAuthnError is an error object from a failed navigator.credentials.get() call.
   * try {
   *   await sdk.reportWebAuthnError({
   *     error: { name: webAuthnError.name, message: webAuthnError.message }
   *   });
   * } catch (submitError) {
   *   console.error("Failed to report WebAuthn error:", submitError);
   * }
   * ```
   */
  reportWebAuthnError(options: ReportWebAuthnErrorOptions): Promise<void>;

  /**
   * Allows the user to opt-out of the desafío de WebAuthn and select a different método MFA.
   * This action submits `action: "pick-authenticator"` to Auth0, which should navigate
   * the user to an pantalla de selección de factor MFA.
   *
   * @param {TryAnotherMethodOptions} [options] - Optional custom parameters for the request.
   * @returns {Promise<void>} A promise that resolves when the action is submitted.
   * @throws {Error} Throws an error if the form submission fails.
   *
   * @example
   * ```typescript
   * // Assuming 'sdk' is an instance of MfaWebAuthnRoamingChallenge
   * try {
   *   await sdk.tryAnotherMethod();
   *   // On success, Auth0 handles redirection to MFA selection.
   * } catch (error) {
   *   console.error("Failed to switch MFA method:", error);
   * }
   * ```
   */
  tryAnotherMethod(options?: TryAnotherMethodOptions): Promise<void>;
}

<ParamField body="reportWebAuthnError" type="Promise<void>">
  Informa a Auth0 de un error específico de la API de WebAuthn.
  Este método debe usarse si `navigator.credentials.get()` falla de una forma que
  `verify()` no gestiona automáticamente, o si el desarrollador quiere
  informar explícitamente de un error antes de intentar otra acción.
  Envía los detalles del error con `action: "showError::{errorDetails}"`.

  Una promesa que se resuelve cuando se envía el informe del error.

  #### Throws

  Lanza un error si falla el envío del formulario.

  ```typescript Example
  // Suponiendo que 'sdk' es una instancia de MfaWebAuthnRoamingChallenge
  // Y webAuthnError es un objeto de error de una llamada fallida a navigator.credentials.get().
  try {
    await sdk.reportWebAuthnError({
      error: { name: webAuthnError.name, message: webAuthnError.message }
    });
  } catch (submitError) {
    console.error("Failed to report WebAuthn error:", submitError);
  }
  ```
