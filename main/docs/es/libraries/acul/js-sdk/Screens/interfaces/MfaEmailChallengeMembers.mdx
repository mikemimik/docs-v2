title: "MfaEmailChallengeMembers"
---

Interfaz que define los métodos y las propiedades disponibles de la pantalla mfa-email-challenge

```ts Example
export interface MfaEmailChallengeMembers extends BaseMembers {
  screen: ScreenMembersOnMfaEmailChallenge;
  untrustedData: UntrustedDataMembersOnMfaEmailChallenge;

  /**
   * Continúa con el desafío por correo electrónico usando el código proporcionado
   * @param payload Las opciones que contienen el código y el indicador rememberDevice
   */
  continue(payload: ContinueOptions): Promise<void>;

  /**
   * Reenvía el código de email
   * @param payload Opciones Custom opcionales para incluir en la request
   */
  resendCode(payload?: ResendCodeOptions): Promise<void>;

  /**
   * Permite al usuario probar otro método de MFA
   * @param payload Opciones Custom opcionales para incluir en la request
   */
  tryAnotherMethod(payload?: TryAnotherMethodOptions): Promise<void>;

  /**
   * Envía la Action para seleccionar una configuración de Email diferente, si está disponible
   * @param payload Opciones Custom opcionales para incluir en la request
   */
  pickEmail(payload?: CustomOptions): Promise<void>;

  /**
   * Obtiene la resend functionality con timeout management para esta screen
   * @param options Opciones de configuración para la resend functionality
   */
  resendManager(options?: StartResendOptions): ResendControl;
}
```
