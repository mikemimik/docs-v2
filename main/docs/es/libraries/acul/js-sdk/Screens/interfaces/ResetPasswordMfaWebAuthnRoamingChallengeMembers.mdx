export interface ResetPasswordMfaWebAuthnRoamingChallengeMembers extends BaseMembers {

  /**
 * Acceso a las propiedades y datos específicos de la pantalla `reset-password-mfa-webauthn-roaming-challenge`,
 * incluyendo las opciones de desafío de `publicKey` de WebAuthn y el indicador `showRememberDevice`.
 * @type {ScreenMembersOnResetPasswordMfaWebAuthnRoamingChallenge}
 */
  screen: ScreenMembersOnResetPasswordMfaWebAuthnRoamingChallenge;

  /**
   * Inicia el proceso de aserción de WebAuthn (desafío de clave de seguridad).
   * Este método llamará internamente a `navigator.credentials.get()` usando el desafío
   * proporcionado en `screen.publicKey`.
   * Tras una interacción exitosa con la clave de seguridad, envía el
   * `PublicKeyCredential` resultante a Auth0 con `action: "default"`.
   *
   * @param {UseSecurityKeyOptions} [options] - Opcional. Parámetros para la operación,
   * como `rememberDevice` (si `screen.showRememberDevice` es true) y otras opciones personalizadas.
   * El campo `response` (la credencial de WebAuthn) es gestionado internamente por el SDK.
   * @returns {Promise<void>} Una promesa que se resuelve cuando el intento de verificación es enviado.
   * Una operación exitosa generalmente resulta en una redirección por parte de Auth0.
   * @throws {Error} Lanza un error si `screen.publicKey` no está disponible, si `navigator.credentials.get()`
   * falla (p. ej., cancelación por parte del usuario, no se encontró ningún autenticador), o si el envío del formulario a Auth0 falla.
   * Se recomienda capturar estos errores y, si corresponde, usar `showError()` para notificar problemas específicos de la API de WebAuthn.
   */
  useSecurityKey(options?: UseSecurityKeyOptions): Promise<void>;

  /**
   * Reporta a Auth0 un error de la API de WebAuthn del lado del cliente (proveniente de `navigator.credentials.get()`).
   * Este método debe usarse cuando `useSecurityKey()` (o una llamada manual a `navigator.credentials.get()`)
   * falla debido a un problema de WebAuthn en el navegador (p. ej., `NotAllowedError` si el usuario cancela el prompt,
   * `NotFoundError` si no se encuentra ningún autenticador coincidente, o un timeout).
   * Envía los detalles del error con `action: "showError::{errorDetailsJsonString}"` y un `response` vacío.
   *
   * @param {ShowErrorOptions} options - Contiene el objeto `error` (con `name` y `message`
   * del DOMException de la API de WebAuthn), un indicador opcional `rememberDevice` y cualquier otra opción personalizada.
   * @returns {Promise<void>} Una promesa que se resuelve cuando el reporte de error es enviado.
   * Auth0 puede volver a renderizar la página con información del error o redirigir al usuario.
   * @throws {Error} Lanza un error si el envío del formulario falla (p. ej., error de red, estado inválido).
   */
  showError(options: ShowErrorOptions): Promise<void>;

  /**
   * Permite al usuario omitir el desafío de WebAuthn itinerante y seleccionar un método de MFA diferente.
   * Esta acción envía `action: "pick-authenticator"` a Auth0, lo que debería dirigir
   * al usuario a una pantalla de selección de factor MFA.
   *
   * @param {TryAnotherMethodOptions} [options] - Opcional. Parámetros para la operación,
   * como `rememberDevice` (si `screen.showRememberDevice` es true) y otras opciones personalizadas.
   * @returns {Promise<void>} Una promesa que se resuelve cuando la acción 'pick-authenticator' es enviada.
   * @throws {Error} Lanza un error si el envío del formulario falla.
   */
  tryAnotherMethod(options?: TryAnotherMethodOptions): Promise<void>;
}

<ParamField body="showError" type="Promise<void>">
  Informa a Auth0 de un error del lado del cliente de la API de WebAuthn (de `navigator.credentials.get()`).
  Este método debe usarse cuando `useSecurityKey()` (o una llamada manual a `navigator.credentials.get()`)
  falla debido a un problema de WebAuthn en el navegador (p. ej., `NotAllowedError` si el usuario cancela el prompt,
  `NotFoundError` si no se encuentra ningún autenticador coincidente, o si se agota el tiempo de espera).
  Envía los detalles del error con `action: "showError::{errorDetailsJsonString}"` y un `response` vacío.

  Una promesa que se resuelve cuando se envía el informe del error.
  Auth0 puede volver a renderizar la página con información del error o redirigir.

  #### Lanza

  Lanza un error si falla el envío del formulario (p. ej., un error de red o `invalid state`).
