Interfaz que define los métodos y las propiedades disponibles de la pantalla mfa-login-options

````ts Example
export interface MfaLoginOptionsMembers extends BaseMembers {
  screen: ScreenMembersOnMfaLoginOptions;
  /**
   * Continúa el proceso de inicio de sesión con el factor MFA seleccionado
   * @param payload Las opciones que contienen el factor seleccionado
   * @example
   * ```typescript
   * const mfaLoginOptions = new MfaLoginOptions();
   * await mfaLoginOptions.enroll({
   *   action: 'push-notification'
   * });
   * ```
   */
  enroll(payload: LoginEnrollOptions): Promise<void>;
  /**
   * Regresa a la pantalla anterior en el flujo de autenticación
   * @param payload Opciones personalizadas opcionales
   * @example
   * ```typescript
   * const mfaLoginOptions = new MfaLoginOptions();
   * await mfaLoginOptions.returnToPrevious();
   * ```
   */
  returnToPrevious(payload?: CustomOptions): Promise<void>;
}
````
