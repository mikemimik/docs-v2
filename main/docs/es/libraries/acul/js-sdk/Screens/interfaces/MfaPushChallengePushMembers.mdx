```ts Example
export interface MfaPushChallengePushMembers extends BaseMembers {
  screen: ScreenMembersOnMfaPushChallengePush;
  untrustedData: UntrustedDataMembersOnMfaPushChallengePush;

  /**
   * Continúa con el desafío de notificación push
   * @param payload Opciones personalizadas opcionales para incluir con la solicitud
   */
  continue(payload?: WithRememberOptions): Promise<void>;

  /**
   * Reenvía la notificación push
   * @param payload Opciones personalizadas opcionales para incluir con la solicitud
   */
  resendPushNotification(payload?: WithRememberOptions): Promise<void>;

  /**
   * Cambia a la introducción manual del código de verificación
   * @param payload Opciones personalizadas opcionales para incluir con la solicitud
   */
  enterCodeManually(payload?: CustomOptions): Promise<void>;

  /**
   * Permite usar otro método de autenticación
   * @param payload Opciones personalizadas opcionales para incluir con la solicitud
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;

  /** 
   * Gestiona el sondeo del estado del desafío push de MFA
   * @param options Opciones de configuración para el sondeo de tipo {@link MfaPollingOptions}
   * @returns Un objeto para controlar el proceso de sondeo
  */
  pollingManager(options: MfaPollingOptions): MfaPushPollingControl;
}
```
