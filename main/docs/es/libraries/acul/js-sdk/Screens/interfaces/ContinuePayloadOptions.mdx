Opciones para continuar con la verificación por correo electrónico

```ts Example
export interface ContinueOptions {
  /** El código ingresado por el usuario */
  code: string;
  /** Indica si se debe recordar el dispositivo */
  rememberDevice?: boolean;
  /** Cualquier opción Custom adicional */
  [key: string]: string | number | boolean | undefined;
}
```
