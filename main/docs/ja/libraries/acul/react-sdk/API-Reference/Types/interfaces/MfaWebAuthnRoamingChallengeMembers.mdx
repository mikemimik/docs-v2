MfaWebAuthnRoamingChallengeMembers

````ts Example
export interface MfaWebAuthnRoamingChallengeMembers extends BaseMembers {
  /**
   * 画面固有のプロパティとデータ（WebAuthn チャレンジオプションを含む）。
   * @type {ScreenMembersOnMfaWebAuthnRoamingChallenge}
   */
  screen: ScreenMembersOnMfaWebAuthnRoamingChallenge;

  /**
   * `screen.data.publicKeyChallengeOptions` で指定されたオプションを使用して `navigator.credentials.get()` を呼び出し、
   * WebAuthn チャレンジを開始します。
   * 成功した場合、`action: "default"` とともに取得した認証情報を Auth0 に送信します。
   * `navigator.credentials.get()` が既知の WebAuthn API エラー（`NotAllowedError` など）で失敗した場合、
   * このメソッドは内部的に `reportWebAuthnError` を呼び出して Auth0 に通知します。
   *
   * @param {VerifySecurityKeyOptions} [options] - 検証プロセスのオプションパラメータ（`rememberDevice` などのカスタムオプションを含む）。
   * @returns {Promise<void>} 検証の試行が送信されると解決される Promise。
   *                          成功した操作は通常リダイレクトになるため、データを直接返しません。
   * @throws {Error} `screen.data.publicKeyChallengeOptions` が存在しない場合、
   *                 `navigator.credentials.get()` が予期しないエラーで失敗した場合、
   *                 または Auth0 へのフォーム送信が失敗した場合にエラーをスローします。
   *
   * @example
   * ```typescript
   * // 'sdk' は MfaWebAuthnRoamingChallenge のインスタンスと仮定します
   * try {
   *   await sdk.verify({ rememberDevice: true });
   *   // 成功時、Auth0 がリダイレクトを処理します。
   * } catch (error) {
   *   console.error("Security key verification failed:", error);
   *   // ページがリロードされた場合、サーバー側の検証メッセージは sdk.transaction.errors を確認してください。
   * }
   * ```
   */
  verify(options?: VerifySecurityKeyOptions): Promise<void>;

  /**
   * 特定の WebAuthn API エラーを Auth0 に報告します。
   * このメソッドは、`verify()` が自動的に処理しない形で `navigator.credentials.get()` が失敗した場合、
   * または別のアクションを試みる前に開発者が明示的にエラーを報告したい場合に使用します。
   * `action: "showError::{errorDetails}"` とともにエラーの詳細を送信します。
   *
   * @param {ReportWebAuthnErrorOptions} options - WebAuthn API からの `name` と `message` を持つ `error` オブジェクト、
   * およびその他のカスタムオプションを含みます。
   * @returns {Promise<void>} エラーレポートが送信されると解決される Promise。
   * @throws {Error} フォームの送信が失敗した場合にエラーをスローします。
   *
   * @example
   * ```typescript
   * // 'sdk' は MfaWebAuthnRoamingChallenge のインスタンスと仮定します
   * // webAuthnError は失敗した navigator.credentials.get() 呼び出しから取得したエラーオブジェクトです。
   * try {
   *   await sdk.reportWebAuthnError({
   *     error: { name: webAuthnError.name, message: webAuthnError.message }
   *   });
   * } catch (submitError) {
   *   console.error("Failed to report WebAuthn error:", submitError);
   * }
   * ```
   */
  reportWebAuthnError(options: ReportWebAuthnErrorOptions): Promise<void>;

  /**
   * ユーザーが WebAuthn チャレンジをスキップして別の MFA メソッドを選択できるようにします。
   * このアクションは `action: "pick-authenticator"` を Auth0 に送信し、
   * ユーザーを MFA 要素選択画面に遷移させます。
   *
   * @param {TryAnotherMethodOptions} [options] - リクエストのカスタムパラメータ（省略可能）。
   * @returns {Promise<void>} アクションが送信されると解決される Promise。
   * @throws {Error} フォームの送信が失敗した場合にエラーをスローします。
   *
   * @example
   * ```typescript
   * // 'sdk' は MfaWebAuthnRoamingChallenge のインスタンスと仮定します
   * try {
   *   await sdk.tryAnotherMethod();
   *   // 成功時、Auth0 が MFA 選択へのリダイレクトを処理します。
   * } catch (error) {
   *   console.error("Failed to switch MFA method:", error);
   * }
   * ```
   */
  tryAnotherMethod(options?: TryAnotherMethodOptions): Promise<void>;
}

<ParamField body="reportWebAuthnError" type="Promise<void>">
  特定の WebAuthn API エラーを Auth0 に報告します。
  このメソッドは、`navigator.credentials.get()` が
  `verify()` で自動処理されない形で失敗した場合や、別の操作を試す前に
  開発者が明示的にエラーを報告したい場合に使用します。
  エラーの詳細は `action: "showError::{errorDetails}"` とともに送信されます。

  エラー報告の送信完了時に解決される Promise です。

  #### Throws

  フォームの送信に失敗した場合はエラーをスローします。

  ```typescript Example
  // 'sdk' が MfaWebAuthnRoamingChallenge のインスタンスであると仮定します
  // また、webAuthnError は失敗した navigator.credentials.get() 呼び出しから返されたエラーオブジェクトです。
  try {
    await sdk.reportWebAuthnError({
      error: { name: webAuthnError.name, message: webAuthnError.message }
    });
  } catch (submitError) {
    console.error("Failed to report WebAuthn error:", submitError);
  }
  ```

<ParamField body="verify" type="Promise<void>">
  `screen.data.publicKeyChallengeOptions` で提供されるオプションを使用して
  `navigator.credentials.get()` を呼び出し、WebAuthn チャレンジを開始します。
  成功した場合は、取得した認証情報を `action: "default"` とともに Auth0 に送信します。
  `navigator.credentials.get()` が既知の WebAuthn API エラー (`NotAllowedError` など) で失敗した場合、
  このメソッドは内部的に `reportWebAuthnError` を呼び出して Auth0 に通知します。

  検証試行の送信完了時に解決される Promise です。
  通常、成功した操作ではリダイレクトが発生するため、データが直接返されることはありません。

  #### Throws

  `screen.data.publicKeyChallengeOptions` が存在しない場合、
  `navigator.credentials.get()` が想定外のエラーで失敗した場合、
  または Auth0 へのフォーム送信に失敗した場合はエラーをスローします。

  ```typescript Example
  // 'sdk' が MfaWebAuthnRoamingChallenge のインスタンスであると仮定します
  try {
    await sdk.verify({ rememberDevice: true });
    // 成功時は、Auth0 がリダイレクトを処理します。
  } catch (error) {
    console.error("Security key verification failed:", error);
    // ページが再読み込みされた場合は、サーバー側の検証メッセージについて sdk.transaction.errors を確認してください。
  }
  ```
