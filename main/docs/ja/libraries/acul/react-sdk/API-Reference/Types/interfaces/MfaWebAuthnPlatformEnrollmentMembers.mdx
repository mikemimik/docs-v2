export interface MfaWebAuthnPlatformEnrollmentMembers extends BaseMembers {
  /**
   * MFA WebAuthn Platform Enrollment固有のデータ構造を持つ画面オブジェクト。
   */

  /**
   * `this.screen.publicKey`で利用可能な公開鍵オプションを使用してWebAuthnプラットフォームの資格情報作成プロセスを開始し、
   * 作成された資格情報をサーバーに送信します。
   * このメソッドは内部的に`createPasskeyCredentials`（`navigator.credentials.create()`のラッパー）を呼び出します。
   *
   * @param {SubmitPasskeyCredentialOptions} [payload] - 作成された資格情報と共にサーバーに送信されるカスタムパラメータ（省略可能）。
   * @returns {Promise<void>} 資格情報が正常に作成・送信されると解決されるPromise。
   * @throws {Error} `this.screen.publicKey`が利用できない場合、`createPasskeyCredentials`が失敗した場合
   * （例：ユーザーによるキャンセル、ハードウェアの問題）、またはサーバーへの送信が失敗した場合にエラーをスローします。
   * @example
   * ```typescript
   * // 'sdk'はMfaWebAuthnPlatformEnrollmentのインスタンスと仮定します
   * try {
   *   if (!sdk.screen.publicKey) { // オプションが利用可能かどうかを確認
   *     throw new Error("Public key creation options are not available on the screen context.");
   *   }
   *   await sdk.submitPasskeyCredential(); // publicKeyを明示的に渡す必要はありません
   *   // 成功した場合、Auth0がリダイレクトを処理します。
   * } catch (error) {
   *   console.error('Passkey enrollment failed:', error);
   *   // エラーを処理します。WebAuthn APIエラーの場合はreportBrowserErrorの呼び出しを検討してください
   *   if (error.name && error.message) { // WebAuthnエラーかどうかを確認
   *     await sdk.reportBrowserError({ error: { name: error.name, message: error.message } });
   *   }
   * }
   * ```
   */
  submitPasskeyCredential(payload?: SubmitPasskeyCredentialOptions): Promise<void>;

  /**

   *

   * } catch (webAuthnError) {

   *      await sdk.reportBrowserError({ error: { name: webAuthnError.name, message: webAuthnError.message } });
   *   }
   *   // UIを更新してユーザーにエラーメッセージを表示します
   * }

<ParamField body="submitPasskeyCredential" type="Promise<void>">
  `this.screen.publicKey` で利用可能な公開鍵オプションを使用して
  WebAuthn プラットフォーム資格情報の作成プロセスを開始し、
  生成された資格情報をサーバーに送信します。
  このメソッドは内部的に `createPasskeyCredentials` (`navigator.credentials.create()` をラップするもの) を呼び出します。

  資格情報の作成と送信が正常に完了すると resolve される Promise。

  #### 例外

  `this.screen.publicKey` が利用できない場合、`createPasskeyCredentials` が失敗した場合
  (例: ユーザーによるキャンセル、ハードウェアの問題) 、またはサーバーへの送信が失敗した場合はエラーをスローします。

  ```typescript Example
  // 'sdk' は MfaWebAuthnPlatformEnrollment のインスタンスであるとします
  try {
    if (!sdk.screen.publicKey) { // オプションが利用可能か確認します
      throw new Error("Public key creation options are not available on the screen context.");
    }
    await sdk.submitPasskeyCredential(); // publicKey を明示的に渡す必要はありません
    // 成功すると、Auth0 がリダイレクトを処理します。
  } catch (error) {
    console.error('Passkey enrollment failed:', error);
    // エラーを処理します。WebAuthn API エラーであれば reportBrowserError を呼び出すこともできます
    if (error.name && error.message) { // WebAuthn エラーのように見えるか確認します
      await sdk.reportBrowserError({ error: { name: error.name, message: error.message } });
    }
  }
  ```
