MFA リカバリーコードチャレンジ画面のメンバーを定義するインターフェース。

```ts Example
export interface ResetPasswordMfaRecoveryCodeChallengeMembers extends BaseMembers {
  client: ClientMembers;
  organization: OrganizationMembers;
  prompt: PromptMembers;
  screen: ScreenMembers;
  transaction: TransactionMembers;

  /**
   * 指定されたリカバリーコードで続行します。
   * @param {string} code - ユーザーが入力したリカバリーコード。
   * @param {CustomOptions} [payload] - 省略可能なペイロード。
   * @returns {Promise<void>}
   */
  continue(code: string, payload?: CustomOptions): Promise<void>;

  /**
   * ユーザーが別のMFA方式を選択できる画面に移動します。
   * @param {CustomOptions} [payload] - 省略可能なペイロード。
   * @returns {Promise<void>}
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;
}
```
