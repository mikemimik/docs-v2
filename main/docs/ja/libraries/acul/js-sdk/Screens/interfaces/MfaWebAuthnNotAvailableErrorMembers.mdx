export interface MfaWebAuthnNotAvailableErrorMembers extends BaseMembers {
  /**
   * WebAuthnが利用できないため、ユーザーが別のMFA方式を選択できるようにします。
   * このactionはユーザーを認証器選択画面に誘導し、利用可能または登録済みの他のMFA要素から選択できます。
   *
   * @param {CustomOptions} [payload] - リクエストとともに送信するオプションのカスタムパラメーター。
   * @returns {Promise<void>} 'pick-authenticator' actionの送信が成功したときに解決されるPromise。
   * @throws {Error} フォームの送信が失敗した場合（例：ネットワークの問題、サーバー側のバリデーションエラー）にエラーをスローします。
   * @example
   * ```typescript
   * // 'sdk' は MfaWebAuthnNotAvailableError のインスタンスであると仮定します
   * try {
   *   await sdk.tryAnotherMethod();
   *   // 成功時、Auth0が認証器選択画面へのリダイレクトを処理します。
   * } catch (error) {
   *   console.error('Failed to switch to another MFA method:', error);
   *   // 必要に応じてUIを更新し、ユーザーにエラーを通知します。
   * }
   * ```
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;
}

<ParamField body="tryAnotherMethod" type="Promise<void>">
  WebAuthn が利用できない場合に、ユーザーが別の MFA方式を選択できるようにします。
  このアクションでは、ユーザーは認証器の選択画面に移動し、
  利用可能な、または登録済みの他の MFA 要素を選択できます。

  &#39;pick-authenticator&#39; アクションが正常に送信されると resolve される Promise です。

  #### 例外

  フォームの送信に失敗した場合 (例: ネットワークの問題、サーバー側のバリデーションエラー) にエラーをスローします。

  ```typescript Example
  // 'sdk' が MfaWebAuthnNotAvailableError のインスタンスであると仮定します
  try {
    await sdk.tryAnotherMethod();
    // 成功すると、Auth0 が認証器選択画面へのリダイレクトを処理します。
  } catch (error) {
    console.error('Failed to switch to another MFA method:', error);
    // 必要に応じて、失敗をユーザーに知らせるように UI を更新します。
  }
  ```
