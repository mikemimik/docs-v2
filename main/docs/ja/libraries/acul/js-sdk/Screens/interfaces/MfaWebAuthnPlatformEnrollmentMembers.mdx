export interface MfaWebAuthnPlatformEnrollmentMembers extends BaseMembers {
  /**
   * MFA WebAuthn Platform Enrollment固有のデータ構造を持つ画面オブジェクト。
   */
  screen: ScreenMembersOnMfaWebAuthnPlatformEnrollment;
  
  /**
   * `this.screen.publicKey`で利用可能な公開鍵オプションを使用してWebAuthnプラットフォームのクレデンシャル作成プロセスを開始し、
   * 作成されたクレデンシャルをサーバーに送信します。
   * このメソッドは内部的に`createPasskeyCredentials`（`navigator.credentials.create()`のラッパー）を呼び出します。
   *
   * @param {SubmitPasskeyCredentialOptions} [payload] - 作成されたクレデンシャルと共にサーバーへ送信されるカスタムパラメータ（省略可能）。
   * @returns {Promise<void>} クレデンシャルが正常に作成・送信されると解決されるPromise。
   * @throws {Error} `this.screen.publicKey`が利用できない場合、`createPasskeyCredentials`が失敗した場合
   * （例：ユーザーによるキャンセル、ハードウェアの問題）、またはサーバーへの送信が失敗した場合にエラーをスローします。
   * @example
   * ```typescript
   * // 'sdk'はMfaWebAuthnPlatformEnrollmentのインスタンスと仮定します
   * try {
   *   if (!sdk.screen.publicKey) { // オプションが利用可能か確認します
   *     throw new Error("Public key creation options are not available on the screen context.");
   *   }
   *   await sdk.submitPasskeyCredential(); // publicKeyを明示的に渡す必要はありません
   *   // 成功時はAuth0がリダイレクトを処理します。
   * } catch (error) {
   *   console.error('Passkey enrollment failed:', error);
   *   // エラーを処理します。WebAuthn APIエラーの場合はreportBrowserErrorの呼び出しを検討してください
   *   if (error.name && error.message) { // WebAuthnエラーかどうかを確認します
   *     await sdk.reportBrowserError({ error: { name: error.name, message: error.message } });
   *   }
   * }
   * ```
   */
  submitPasskeyCredential(payload?: SubmitPasskeyCredentialOptions): Promise<void>;

  /**
   * WebAuthnの`navigator.credentials.create()`操作中に発生したブラウザ側のエラーを報告します。
   * このメソッドはエラーの詳細をサーバーに送信します。
   *
   * @param {ReportBrowserErrorOptions} payload - ブラウザのエラー詳細およびカスタムオプション。
   * @returns {Promise<void>} エラーレポートが正常に送信されると解決されるPromise。
   * @throws {Error} 送信が失敗した場合にエラーをスローします。
   * @example
   * ```typescript
   * // 'sdk'はMfaWebAuthnPlatformEnrollmentのインスタンスと仮定します
   * // パスキー作成試行のcatchブロック内（例：submitPasskeyCredentialから）：
   * } catch (webAuthnError) {
   *   if (webAuthnError.name) { // WebAuthn APIエラーである可能性を確認します
   *      await sdk.reportBrowserError({ error: { name: webAuthnError.name, message: webAuthnError.message } });
   *   }
   *   // エラーメッセージをユーザーに表示するようUIを更新します
   * }
````

<ParamField body="submitPasskeyCredential" type="Promise<void>">
  `this.screen.publicKey` で利用可能な公開鍵オプションを使用して
  WebAuthn プラットフォームのクレデンシャル作成プロセスを開始し、
  生成されたクレデンシャルをサーバーに送信します。
  このメソッドは内部で `createPasskeyCredentials` (`navigator.credentials.create()` をラップしたもの) を呼び出します。

  クレデンシャルが正常に作成され、送信されると解決される Promise です。

  #### Throws

  `this.screen.publicKey` が利用できない場合、`createPasskeyCredentials` が失敗した場合
  (例: ユーザーによるキャンセル、ハードウェアの問題) 、またはサーバーへの送信が失敗した場合に
  エラーをスローします。

  ```typescript Example
  // 'sdk' が MfaWebAuthnPlatformEnrollment のインスタンスであると仮定します
  try {
    if (!sdk.screen.publicKey) { // オプションが利用可能か確認します
      throw new Error("Public key creation options are not available on the screen context.");
    }
    await sdk.submitPasskeyCredential(); // publicKey を明示的に渡す必要はありません
    // 成功時は、Auth0 がリダイレクトを処理します。
  } catch (error) {
    console.error('Passkey enrollment failed:', error);
    // エラーを処理します。必要に応じて、WebAuthn API エラーの場合は reportBrowserError を呼び出します
    if (error.name && error.message) { // WebAuthn エラーのように見えるか確認します
      await sdk.reportBrowserError({ error: { name: error.name, message: error.message } });
    }
  }
  ```
