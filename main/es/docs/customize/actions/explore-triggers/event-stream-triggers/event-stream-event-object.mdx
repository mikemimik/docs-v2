<ResponseField name="event.message" type="dictionary">

      Identifica el evento.

    <ResponseField name="type" type="string">

    </ResponseField>

    <ResponseField name="data" type="dictionary" post={["optional"]}>
      El payload del evento.
    </ResponseField>

    <ResponseField name="source" type="string">
      Identifica el contexto en el que ocurrió el evento.
    </ResponseField>

    <ResponseField name="specversion" type="string">
      La versión de la especificación de CloudEvents que usa el evento.
    </ResponseField>

    <ResponseField name="time" type="string" post={["optional"]}>

    </ResponseField>
  </Expandable>
