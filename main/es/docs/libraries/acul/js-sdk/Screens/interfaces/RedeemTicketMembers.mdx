---
title: "RedeemTicketMembers"
---

Interfaz que describe los datos disponibles en la pantalla «Redeem Ticket».

````ts Example
export interface RedeemTicketMembers extends BaseMembers {
  client: ClientMembers;
  organization: OrganizationMembers;
  prompt: PromptMembers;
  screen: ScreenMembers;
  transaction: TransactionMembers;

  /**
   * Realiza la acción predeterminada en la pantalla redeem-ticket, que generalmente implica continuar el flujo.
   * @param {CustomOptions} [payload] - Opciones personalizadas opcionales para incluir en la solicitud.
   * @returns {Promise<void>} Una promesa que se resuelve cuando la acción se completa.
   * @throws {Error} Si la operación falla.
   * @example
   * ```typescript
   * import RedeemTicket from '@auth0/auth0-acul-js/redeem-ticket';
   *
   * const redeemTicket = new RedeemTicket();
   * await redeemTicket.continue();
   * ```
   */
  continue(payload?: CustomOptions): Promise<void>;
}
````
