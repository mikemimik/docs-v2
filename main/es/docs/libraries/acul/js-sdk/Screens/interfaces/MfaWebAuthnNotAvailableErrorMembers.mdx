export interface MfaWebAuthnNotAvailableErrorMembers extends BaseMembers {

   * Permite al usuario seleccionar un método de MFA diferente porque WebAuthn no está disponible.

   * @param {CustomOptions} [payload] - Parámetros personalizados opcionales que se enviarán con la solicitud.

   * ```typescript
   * // Se asume que 'sdk' es una instancia de MfaWebAuthnNotAvailableError
   * try {
   *   await sdk.tryAnotherMethod();
   *   // En caso de éxito, Auth0 gestiona la redirección a la pantalla de selección de autenticador.
   * } catch (error) {
   *   console.error('Failed to switch to another MFA method:', error);
   *   // Actualizar la interfaz para informar al usuario del error si es necesario.
   * }
   * ```
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;

<ParamField body="tryAnotherMethod" type="Promise<void>">
  Permite al usuario seleccionar otro método de MFA porque WebAuthn no está disponible.
  Esta Action dirige al usuario a una pantalla de selección de autenticador, donde puede elegir
  entre otros factores de MFA disponibles o ya registrados.

  Una Promise que se resuelve cuando la Action `pick-authenticator` se envía correctamente.

  #### Lanza

  Lanza un error si falla el envío del formulario (por ejemplo, por problemas de red o errores de validación del lado del servidor).

  ```typescript Example
  // Suponiendo que 'sdk' es una instancia de MfaWebAuthnNotAvailableError
  try {
    await sdk.tryAnotherMethod();
    // Si se realiza correctamente, Auth0 se encarga de redirigir a la pantalla de selección de autenticador.
  } catch (error) {
    console.error('Failed to switch to another MFA method:', error);
    // Opcionalmente, actualiza la UI para informar al usuario del fallo.
  }
  ```
