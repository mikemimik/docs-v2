export interface MfaWebAuthnPlatformChallengeMembers extends BaseMembers {
  /**
   * Screen-specific properties and data, including WebAuthn challenge options and remember device preference.
   * @type {ScreenMembersOnMfaWebAuthnPlatformChallenge}
   */
  screen: ScreenMembersOnMfaWebAuthnPlatformChallenge;

  /**
   * Inicia el challenge del autenticador de plataforma WebAuthn.
   * Este método llama internamente a `navigator.credentials.get()` usando las opciones de challenge
   * proporcionadas en `screen.publicKey`.
   * Si tiene éxito, envía las credenciales resultantes a Auth0 con `action: "default"`.
   *
   * Si `navigator.credentials.get()` falla (por ejemplo, cancelación por parte del usuario, tiempo de espera u otros errores de la API de WebAuthn),
   * se recomienda llamar a `reportBrowserError` con los detalles del error.
   *
   * @param {VerifyPlatformAuthenticatorOptions} [options] - Parámetros opcionales para el proceso de verificación,
   * como `rememberDevice` (si `screen.showRememberDevice` es true) y otras opciones personalizadas.
   * @returns {Promise<void>} Una promesa que se resuelve cuando se envía el intento de verificación.
   *                          Una operación exitosa generalmente resulta en una redirección.
   * @throws {Error} Lanza un error si `screen.publicKey` no está presente,
   *                 si `navigator.credentials.get()` falla con un error inesperado no gestionado por `getPasskeyCredentials`,
   *                 o si el envío del formulario a Auth0 falla.
   *
   * @example
   * ```typescript
   * // Assuming 'sdk' is an instance of MfaWebAuthnPlatformChallenge
   * try {
   *   const remember = sdk.screen.showRememberDevice && userCheckedRememberDeviceBox;
   *   await sdk.verify({ rememberDevice: remember });
   *   // On success, Auth0 handles redirection.
   * } catch (error) {
   *   console.error("Platform authenticator verification failed:", error);
   *   // If it's a WebAuthn API error, report it
   *   if (error.name && error.message) { // Basic check for DOMException like error
   *     await sdk.reportBrowserError({ error: { name: error.name, message: error.message } });
   *   }
   *   // Check sdk.transaction.errors for server-side validation messages if the page reloads.
   * }
   * ```
   */
  verify(options?: VerifyPlatformAuthenticatorOptions): Promise<void>;

  /**
   * Reports a specific WebAuthn API error (from `navigator.credentials.get()`) to Auth0.
   * This method should be used when `verify()` (or a manual `navigator.credentials.get()` call) fails due to
   * a browser-side WebAuthn issue (e.g., user cancellation `NotAllowedError`, timeout).
   * It submits the error details with `action: "showError::{errorDetailsJsonString}"` and an empty `response`.
   *
   * @param {ReportBrowserErrorOptions} options - Contains the `error` object (with `name` and `message`
   * from the WebAuthn API DOMException) and any other custom options.
   * @returns {Promise<void>} A promise that resolves when the error report is submitted.
   * @throws {Error} Throws an error if the form submission fails (e.g., network error, invalid state).
   *
   * @example
   * ```typescript
   * // Assuming 'sdk' is an instance of MfaWebAuthnPlatformChallenge
   * // And webAuthnDomException is an error object from a failed navigator.credentials.get() call.
   * try {
   *   await sdk.reportBrowserError({
   *     error: { name: webAuthnDomException.name, message: webAuthnDomException.message }
   *   });
   *   // Auth0 will process this error and may re-render the page or redirect.
   * } catch (submitError) {
   *   console.error("Failed to report WebAuthn browser error:", submitError);
   * }
   * ```
   */
  reportBrowserError(options: ReportBrowserErrorOptions): Promise<void>;

  /**
   * Allows the user to opt-out of the WebAuthn platform challenge and select a different MFA method.
   * This action submits `action: "pick-authenticator"` to Auth0, which should navigate
   * the user to an MFA factor selection screen.
   *
   * @param {TryAnotherMethodOptions} [options] - Optional custom parameters to be sent with the request.
   * @returns {Promise<void>} A promise that resolves when the 'pick-authenticator' action is submitted.
   * @throws {Error} Throws an error if the form submission fails.
   *
   * @example
   * ```typescript
   * // Assuming 'sdk' is an instance of MfaWebAuthnPlatformChallenge
   * try {
   *   await sdk.tryAnotherMethod();
   *   // On success, Auth0 handles redirection to MFA selection.
   * } catch (error) {
   *   console.error("Failed to switch MFA method:", error);
   * }
   * ```
   */
  tryAnotherMethod(options?: TryAnotherMethodOptions): Promise<void>;
}

<ParamField body="reportBrowserError" type="Promise<void>">
  Informa a Auth0 sobre un error específico de la API de WebAuthn (de `navigator.credentials.get()`).
  Este método debe usarse cuando `verify()` (o una llamada manual a `navigator.credentials.get()`) falla debido a
  un problema de WebAuthn en el navegador (p. ej., cancelación del usuario `NotAllowedError`, tiempo de espera agotado).
  Envía los detalles del error con `action: "showError::{errorDetailsJsonString}"` y una `response` vacía.

  Una promesa que se resuelve cuando se envía el informe de error.

  #### Lanza

  ```typescript Example
  // Y que webAuthnDomException es un objeto de error de una llamada fallida a navigator.credentials.get().

      error: { name: webAuthnDomException.name, message: webAuthnDomException.message }

    // Auth0 procesará este error y puede volver a renderizar la página o redirigir.
  } catch (submitError) {
    console.error("Failed to report WebAuthn browser error:", submitError);
  }
  ```

<ParamField body="verify" type="Promise<void>">
  Inicia el desafío del autenticador de plataforma de WebAuthn.
  Este método llama internamente a `navigator.credentials.get()` con las opciones del desafío
  proporcionadas en `screen.publicKey`.
  Si tiene éxito, envía la credencial resultante a Auth0 con `action: "default"`.

  Si `navigator.credentials.get()` falla (p. ej., por cancelación del usuario, tiempo de espera agotado u otros errores de la API de WebAuthn),
  se recomienda llamar a `reportBrowserError` con los detalles del error.

  Una promesa que se resuelve cuando se envía el intento de verificación.
  Una operación exitosa normalmente provoca una redirección.

  #### Lanza

  Lanza un error si falta `screen.publicKey`,
  si `navigator.credentials.get()` falla con un error inesperado no controlado por `getPasskeyCredentials`,
  o si falla el envío del formulario a Auth0.

  ```typescript Example
  // Suponiendo que 'sdk' es una instancia de MfaWebAuthnPlatformChallenge
  try {
    const remember = sdk.screen.showRememberDevice && userCheckedRememberDeviceBox;
    await sdk.verify({ rememberDevice: remember });
    // Si tiene éxito, Auth0 se encarga de la redirección.
  } catch (error) {
    console.error("Platform authenticator verification failed:", error);
    // Si es un error de la API de WebAuthn, repórtalo
    if (error.name && error.message) { // Verificación básica para un error de tipo DOMException
      await sdk.reportBrowserError({ error: { name: error.name, message: error.message } });
    }
    // Revisa sdk.transaction.errors para ver los mensajes de validación del lado del servidor si la página se recarga.
  }
  ```
