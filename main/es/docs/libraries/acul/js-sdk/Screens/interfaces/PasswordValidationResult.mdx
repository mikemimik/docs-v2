---
title: "PasswordValidationResult"
---

```ts Example
export interface PasswordValidationResult {
  /**
   * Resultado general de todas las verificaciones de política de contraseña.
   *
   * `true` solo si cada regla en {@link PasswordComplexityRule} tiene `status` igual a `"valid"`.
   */
  isValid: boolean;

  /**
   * Lista de reglas de contraseña evaluadas.
   *
   * Cada elemento es un {@link PasswordComplexityRule} cuyo campo
   * {@link PasswordComplexityRule.status | status} refleja el
   * resultado de validación calculado para esa regla.
   */
  results: PasswordComplexityRule[];
}
```

<div id="properties">
  ## Propiedades
</div>

<ParamField body="isValid" type="boolean">
  Resultado global de todas las comprobaciones de la política de contraseñas.

  `true` solo si todas las reglas de [PasswordComplexityRule](/es/docs/libraries/acul/js-sdk/Screens/interfaces/PasswordComplexityRule) tienen `status` igual a `"valid"`.
</ParamField>

<ParamField body="results" type={<span><a href="/es/docs/libraries/acul/js-sdk/Screens/interfaces/PasswordComplexityRule">PasswordComplexityRule</a></span>}>
  Lista de reglas de contraseña evaluadas.

  Cada elemento es una [PasswordComplexityRule](/es/docs/libraries/acul/js-sdk/Screens/interfaces/PasswordComplexityRule) cuyo
  campo [status](/es/docs/libraries/acul/js-sdk/Screens/interfaces/PasswordComplexityRule#status) refleja el
  resultado de validación calculado para esa regla.
</ParamField>
