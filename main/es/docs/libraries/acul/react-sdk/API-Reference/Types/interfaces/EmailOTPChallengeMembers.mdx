---
title: "Members del desafío OTP de Email"
---

Representa la interfaz de la pantalla Members del desafío OTP de Email.

```ts Example
export interface EmailOTPChallengeMembers extends BaseMembers {
  screen: ScreenMembersOnEmailOTPChallenge;
  /**
   * Envía el código OTP ingresado por el usuario.
   * @param options Parámetros opcionales para incluir en el envío.
   */
  submitCode(options: OtpCodeOptions): Promise<void>;
  /**
   * Solicita que se envíe un nuevo código OTP al correo electrónico del usuario.
   * @param options Parámetros opcionales para incluir en la solicitud de reenvío.
   */
  resendCode(options?: CustomOptions): Promise<void>;
  /**
   * Obtiene la funcionalidad de reenvío con gestión del tiempo de espera para esta pantalla.
   * @param options Opciones de configuración para la funcionalidad de reenvío.
   */
  resendManager(options?: StartResendOptions): ResendControl;
}
```
