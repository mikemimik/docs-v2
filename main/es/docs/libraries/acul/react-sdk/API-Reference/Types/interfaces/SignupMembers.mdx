````ts Example
export interface SignupMembers extends BaseMembers {
  screen: ScreenMembersOnSignup;
  transaction: TransactionMembersOnSignup;
  signup(payload: SignupOptions): Promise<void>;
  federatedSignup(payload: FederatedSignupOptions): Promise<void>;
  pickCountryCode(payload?: CustomOptions): Promise<void>;
  validatePassword(password: string): PasswordValidationResult;
  /**
 * Devuelve una lista de identificadores habilitados (p. ej., email, phone, username)
 * según el estado actual de la transacción.
 *
 * Los identificadores pueden ser obligatorios u opcionales según la estrategia de conexión
 * y la configuración proporcionada durante el flujo de autenticación o de registro.
 *
 * @returns Un arreglo de objetos {@link Identifier} habilitados, o `null` si la transacción no está inicializada.
 *
 * @example
 * ```ts
 * const identifiers = authClient.getSignupIdentifiers();
 * if (identifiers) {
 *   identifiers.forEach(({ type, required }) => {
 *     console.log(`${type} is ${required ? 'required' : 'optional'}`);
 *   });
 * }
 * ```
 * @utilityFeature
 * @see Identifier
 */
  getSignupIdentifiers(): Identifier[] | null;
  validateUsername(username: string): UsernameValidationResult;
}
````

<ParamField body="getSignupIdentifiers" type={<span><a href="/es/docs/libraries/acul/react-sdk/API-Reference/Types/interfaces/Identifier">Identifier</a>[]</span>}>
  **`Función de utilidad`**

  Devuelve una lista de identificadores habilitados (p. ej., email, phone, username)
  en función del estado actual de la transacción.

  Los identificadores pueden ser obligatorios u opcionales según la estrategia de conexión
  y la configuración proporcionada durante el flujo de autenticación o de registro.

  Un arreglo de objetos [Identifier](/es/docs/libraries/acul/react-sdk/API-Reference/Types/interfaces/Identifier) habilitados, o `null` si la transacción no está inicializada.

  ```ts Example
  const identifiers = authClient.getSignupIdentifiers();
  if (identifiers) {
    identifiers.forEach(({ type, required }) => {
      console.log(`${type} is ${required ? 'required' : 'optional'}`);
    });
  }
  ```
</ParamField>
