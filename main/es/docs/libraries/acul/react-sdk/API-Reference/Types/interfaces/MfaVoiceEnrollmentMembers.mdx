---
title: "MfaVoiceEnrollmentMembers"
---

Interfaz que describe los miembros de la pantalla de registro por voz de MFA.

```ts Example
export interface MfaVoiceEnrollmentMembers extends BaseMembers {
  client: ClientMembers;
  organization: OrganizationMembers;
  prompt: PromptMembers;
  screen: ScreenMembers;
  transaction: TransactionMembers;

  /**
   * Continúa con el proceso de registro por voz.
   * @param payload - El número de teléfono y las opciones personalizadas opcionales.
   * @returns Promise que se resuelve cuando el registro continúa.
   */
  continue(payload: ContinueOptions): Promise<void>;

  /**
   * Permite probar otro método de autenticación.
   * @param payload - Opciones personalizadas opcionales.
   * @returns Promise que se resuelve cuando el usuario cambia de método.
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;

  /**
   * Permite seleccionar un código de país para el número de teléfono.
   * @param payload - Opciones personalizadas opcionales.
   * @returns Promise que se resuelve cuando se selecciona el código de país.
   */
  selectPhoneCountryCode(payload?: CustomOptions): Promise<void>;
}
```
