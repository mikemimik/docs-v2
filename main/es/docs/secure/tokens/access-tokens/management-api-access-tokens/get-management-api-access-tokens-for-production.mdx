Para realizar llamadas programadas frecuentes en un entorno de producción, debes crear un proceso en tu backend que te proporcione automáticamente un token (y así simular un token que no caduca).

<div id="prerequisites">
  ## Requisitos previos
</div>

* [Registrar aplicaciones de máquina a máquina](/docs/es/get-started/auth0-overview/create-applications/machine-to-machine-apps).

<div id="get-access-tokens">
  ## Obtener tokens de acceso
</div>

Para solicitar a Auth0 un token v2 de <Tooltip tip="Management API: Un producto que permite a los clientes realizar tareas administrativas." cta="Ver glosario" href="/docs/es/glossary?term=Management+API">Management API</Tooltip>, realiza una operación `POST` en el endpoint `https://{yourDomain}/oauth/token` con las credenciales de la Machine-to-Machine Application que creaste en el paso previo.

El payload debe tener el siguiente formato:

<AuthCodeGroup>
  ```bash cURL
  curl --request POST \
    --url 'https://{yourDomain}/oauth/token' \
    --header 'content-type: application/x-www-form-urlencoded' \
    --data grant_type=client_credentials \
    --data 'client_id={yourClientId}' \
    --data 'client_secret={yourClientSecret}' \
    --data 'audience=https://{yourDomain}/api/v2/'
  ```

  ```csharp C#
  var client = new RestClient("https://{yourDomain}/oauth/token");
  var request = new RestRequest(Method.POST);
  request.AddHeader("content-type", "application/x-www-form-urlencoded");
  request.AddParameter("application/x-www-form-urlencoded", "grant_type=client_credentials&client_id={yourClientId}&client_secret=%7ByourClientSecret%7D&audience=https%3A%2F%2F{yourDomain}%2Fapi%2Fv2%2F", ParameterType.RequestBody);
  IRestResponse response = client.Execute(request);
  ```

  ```go Go
  package main

  import (
  	"fmt"
  	"strings"
  	"net/http"
  	"io/ioutil"
  )

  func main() {

  	url := "https://{yourDomain}/oauth/token"

  	payload := strings.NewReader("grant_type=client_credentials&client_id={yourClientId}&client_secret=%7ByourClientSecret%7D&audience=https%3A%2F%2F{yourDomain}%2Fapi%2Fv2%2F")

  	req, _ := http.NewRequest("POST", url, payload)

  	req.Header.Add("content-type", "application/x-www-form-urlencoded")

  	res, _ := http.DefaultClient.Do(req)

  	defer res.Body.Close()
  	body, _ := ioutil.ReadAll(res.Body)

  	fmt.Println(res)
  	fmt.Println(string(body))

  }
  ```

  ```java Java
  HttpResponse<String> response = Unirest.post("https://{yourDomain}/oauth/token")
    .header("content-type", "application/x-www-form-urlencoded")
    .body("grant_type=client_credentials&client_id={yourClientId}&client_secret=%7ByourClientSecret%7D&audience=https%3A%2F%2F{yourDomain}%2Fapi%2Fv2%2F")
    .asString();
  ```

  ```javascript Node.JS
  var axios = require("axios").default;

  var options = {
    method: 'POST',
    url: 'https://{yourDomain}/oauth/token',
    headers: {'content-type': 'application/x-www-form-urlencoded'},
    data: new URLSearchParams({
      grant_type: 'client_credentials',
      client_id: '{yourClientId}',
      client_secret: '{yourClientSecret}',
      audience: 'https://{yourDomain}/api/v2/'
    })
  };

  axios.request(options).then(function (response) {
    console.log(response.data);
  }).catch(function (error) {
    console.error(error);
  });
  ```

  ```php PHP
  $curl = curl_init();

  curl_setopt_array($curl, [
    CURLOPT_URL => "https://{yourDomain}/oauth/token",
    CURLOPT_RETURNTRANSFER => true,
    CURLOPT_ENCODING => "",
    CURLOPT_MAXREDIRS => 10,
    CURLOPT_TIMEOUT => 30,
    CURLOPT_HTTP_VERSION => CURL_HTTP_VERSION_1_1,
    CURLOPT_CUSTOMREQUEST => "POST",
    CURLOPT_POSTFIELDS => "grant_type=client_credentials&client_id={yourClientId}&client_secret=%7ByourClientSecret%7D&audience=https%3A%2F%2F{yourDomain}%2Fapi%2Fv2%2F",
    CURLOPT_HTTPHEADER => [
      "content-type: application/x-www-form-urlencoded"
    ],
  ]);

  $response = curl_exec($curl);
  $err = curl_error($curl);

  curl_close($curl);

  if ($err) {
    echo "cURL Error #:" . $err;
  } else {
    echo $response;
  }
  ```

  ```python Python
  import http.client

  conn = http.client.HTTPSConnection("")

  payload = "grant_type=client_credentials&client_id={yourClientId}&client_secret=%7ByourClientSecret%7D&audience=https%3A%2F%2F{yourDomain}%2Fapi%2Fv2%2F"

  headers = { 'content-type': "application/x-www-form-urlencoded" }

  conn.request("POST", "/{yourDomain}/oauth/token", payload, headers)

  res = conn.getresponse()
  data = res.read()

  print(data.decode("utf-8"))
  ```

  ```ruby Ruby
  require 'uri'
  require 'net/http'
  require 'openssl'

  url = URI("https://{yourDomain}/oauth/token")

  http = Net::HTTP.new(url.host, url.port)
  http.use_ssl = true
  http.verify_mode = OpenSSL::SSL::VERIFY_NONE

  request = Net::HTTP::Post.new(url)
  request["content-type"] = 'application/x-www-form-urlencoded'
  request.body = "grant_type=client_credentials&client_id={yourClientId}&client_secret=%7ByourClientSecret%7D&audience=https%3A%2F%2F{yourDomain}%2Fapi%2Fv2%2F"

  response = http.request(request)
  puts response.read_body
  ```
</AuthCodeGroup>

Recuerda actualizar `` `{yourClientSecret}` `` con el <Tooltip tip="Client Secret: Secreto que usa un cliente (aplicación) para autenticarse con el servidor de autorización; solo deben conocerlo el cliente y el servidor de autorización, y debe ser lo suficientemente aleatorio para que no se pueda adivinar." cta="Ver glosario" href="/docs/es/glossary?term=client+secret">secreto del cliente</Tooltip> en la [pestaña Settings de tu Application](https://manage.auth0.com/#/applications/\{yourClientId}/settings).

Los parámetros de la solicitud son:

| Parámetro de la solicitud | Descripción                                                                                                                                                                                                                     |
| ------------------------- | ------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------- |
| **grant&#95;type**        | Indica qué [flujo de OAuth 2.0](/docs/es/authenticate/protocols/oauth#grant-types) quieres ejecutar. Para la comunicación de máquina a máquina, usa el valor `client_credentials`.                                                 |
| **client&#95;id**         | Este es el valor del campo **Client ID** de la Machine-to-Machine Application que creaste. Puedes encontrarlo en la [pestaña Settings de tu Application](https://manage.auth0.com/#/applications/\{yourClientId}/settings).     |
| **client&#95;secret**     | Este es el valor del campo **Client Secret** de la Machine-to-Machine Application que creaste. Puedes encontrarlo en la [pestaña Settings de tu Application](https://manage.auth0.com/#/applications/\{yourClientId}/settings). |
| **audience**              | Este es el valor del campo **Identifier** de la `Auth0 Management API`. Puedes encontrarlo en la [pestaña Settings de la API](https://manage.auth0.com/#/apis).                                                                 |

<Warning>
  Usa los scopes `update:client_grants` y `create:client_grants` solo con aplicaciones de alto privilegio, ya que permiten que el cliente se otorgue a sí mismo más permisos.
</Warning>

La respuesta contendrá un [JWT firmado](/docs/es/secure/tokens/json-web-tokens), un tiempo de expiración, los scopes concedidos y el tipo de token.

```json lines
{
  "access_token": "eyJ...Ggg",
  "expires_in": 86400,
  "scope": "read:clients create:clients read:client_keys",
  "token_type": "Bearer"
}
```

De lo anterior, podemos ver que nuestro <Tooltip tip="Token de acceso: credencial de autorización, en forma de una cadena opaca o JWT, utilizada para acceder a una API." cta="Ver glosario" href="/docs/es/glossary?term=Access+Token">Token de acceso</Tooltip> expirará en 24 horas (86 400 segundos), está autorizado para leer y crear aplicaciones, y es un [Bearer Access Token](https://tools.ietf.org/html/rfc6750).

<div id="example-python-implementation">
  ## Ejemplo: implementación en Python
</div>

Este script de Python obtiene un token de acceso para la Management API v2, lo usa para llamar al endpoint [obtener](https://auth0.com/docs/api/management/v2#!/Clients/get_clients) e imprime la respuesta en la consola.

Antes de ejecutarlo, asegúrate de que las siguientes variables contengan valores válidos:

* `AUDIENCE`: El **Identifier** de la `Auth0 Management API`. Puedes encontrarlo en la [pestaña Settings de la API](https://manage.auth0.com/#/apis).
* `DOMAIN`: El **Domain** de la Machine-to-Machine Application que creaste.
* `CLIENT_ID`: El **<Tooltip tip="Client ID: Valor de identificación que Auth0 asigna a tu recurso registrado." cta="Ver glosario" href="/docs/es/glossary?term=Client+ID">Client ID</Tooltip>** de la Machine-to-Machine Application que creaste.
* `CLIENT_SECRET`: El **Client Secret** de la Machine-to-Machine Application que creaste.

```python lines expandable
def main():
  import json, requests
  from requests.exceptions import RequestException, HTTPError, URLRequired

  # Valores de configuración
  domain = 'YOUR_DOMAIN'
  audience = f'https://{domain}/api/v2/'
  client_id = '{yourClientId}'
  client_secret = '{yourClientSecret}'
  grant_type = "client_credentials" # Flujo de OAuth 2.0 a usar

  # Obtener un token de acceso de Auth0
  base_url = f"https://{domain}"
  payload =  { 
    'grant_type': grant_type,
    'client_id': client_id,
    'client_secret': client_secret,
    'audience': audience
  }
  response = requests.post(f'{base_url}/oauth/token', data=payload)
  oauth = response.json()
  access_token = oauth.get('access_token')

  # Agregar el token al encabezado Authorization de la solicitud
  headers = {
    'Authorization': f'Bearer {access_token}',
    'Content-Type': 'application/json'
  }

  # Obtener todas las aplicaciones con el token
  try:
    res = requests.get(f'{base_url}/api/v2/clients', headers=headers)
    print(res.json())
  except HTTPError as e:
    print(f'HTTPError: {str(e.code)} {str(e.reason)}')
  except URLRequired as e:
    print(f'URLRequired: {str(e.reason)}')
  except RequestException as e:
    print(f'RequestException: {e}')
  except Exception as e:
    print(f'Generic Exception: {e}')

# Código estándar para llamar a la función main().
if __name__ == '__main__':
  main()
```
