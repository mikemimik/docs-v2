import 'package:auth0_flutter/auth0_flutter.dart';
import 'package:flutter_dotenv/flutter_dotenv.dart';

class AuthService {
  late final Auth0 _auth0;
  Credentials? _credentials;

  AuthService() {
    _auth0 = Auth0(
      dotenv.env['AUTH0_DOMAIN']!,    // アプリケーション設定 → Domain から取得
      dotenv.env['AUTH0_CLIENT_ID']!, // アプリケーション設定 → Client ID から取得
    );
  }

  Credentials? get credentials => _credentials;
  bool get isAuthenticated => _credentials != null;

  Future<void> login() async {
    final result = await _auth0.windowsWebAuthentication().login(
      appCustomURL: 'auth0flutter://callback', // 登録済みスキームと一致させる必要があります
      authTimeout: const Duration(minutes: 5),
    );
    _credentials = result;
  }

  Future<void> logout() async {
    await _auth0.windowsWebAuthentication().logout(
      appCustomURL: 'auth0flutter://callback',
    );
    _credentials = null;
  }
}
