<?php

  declare(strict_types=1);

  /**
   * Auth0 Universal Login から戻った後、`exchange()` メソッドを使用してコード交換を行い、
   * 認証フローを完了する必要があります。このプロセスにより、アプリケーションが使用するセッションが設定されます。
   *
   * 成功した場合、ユーザーはインデックスルートにリダイレクトされます。
   */

   $hasAuthenticated = isset($_GET['state']) && isset($_GET['code']);
   $hasAuthenticationFailure = isset($_GET['error']);

  // 成功した場合、エンドユーザーのリクエストには ?state および ?code の値が含まれます。
  if ($hasAuthenticated) {
    try {
      $sdk->exchange();
    } catch (\Throwable $th) {
      printf('Unable to complete authentication: %s', $th->getMessage());
      exit;
    }
  }

  // 認証が失敗した場合、エンドユーザーのリクエストには ?error が含まれます。
  if ($hasAuthenticationFailure) {
    printf('Authentication failure: %s', htmlspecialchars(strip_tags(filter_input(INPUT_GET, 'error'))));
    exit;
  }

  // 処理なし: インデックスルートにリダイレクトします。
  header('Location: /');
