private func fetchSessionToken(accessToken: AccessToken, completion: @escaping (String?, Error?) -> Void) {
    let request = GraphRequest(graphPath: "oauth/token")
    request.parameters["grant_type"] = "fb_attenuate_token"
    request.parameters["client_id"] = plistValues(bundle: Bundle.main)
    request.parameters["fb_exchange_token"] = accessToken.tokenString
    request.start { connection, response, error in
        if let error = error {
            completion(nil, error)
            return
        }
        if let responseDict = response as? [String: Any] {
            if let accessToken = responseDict["access_token"] as? String {
                completion(accessToken, nil)
            }
        }
    }
}

private func plistValues(bundle: Bundle) -> String? {
    guard let appId = bundle.infoDictionary?["FacebookAppID"] as? String else {
        return nil
    }
    return appId
}
