private lateinit var fbCallbackManager: CallbackManager
private val account: Auth0 by lazy {
    // -- Remplacez ces identifiants par ceux de votre application Auth0 !
    val account = Auth0.getInstance(
        getString(R.string.com_auth0_client_id),
        getString(R.string.com_auth0_domain)
    )
    // N'activez pas la journalisation du trafic réseau en production !
    account.networkingClient = DefaultClient(enableLogging = true)
    account
}

override fun onCreate(savedInstanceState: Bundle?) {
    super.onCreate(savedInstanceState)
    setContentView(R.layout.activity_login)
    fbCallbackManager = CallbackManager.Factory.create()
    val loginButton = findViewById<LoginButton>(R.id.login_button)
    loginButton.setPermissions("email", "public_profile")
    loginButton.registerCallback(fbCallbackManager,
        object : FacebookCallback<LoginResult> {
        override fun onSuccess(result: LoginResult) {
            val accessToken = result.accessToken
            performLogin(accessToken)
        }
        override fun onCancel() {
            //L'utilisateur a fermé la boîte de dialogue. Vous pouvez ignorer ceci.
        }
        override fun onError(error: FacebookException) {
            //Gérer l'erreur d'authentification Facebook
        }
    })
}
