// Importer les plugins d'application et de navigateur de Capacitor pour accéder à `addListener`, `appUrlOpen`,
// ainsi qu'aux éléments nécessaires pour Auth0 et React
import { App as CapApp } from '@capacitor/app';
import { Browser } from '@capacitor/browser';
import { useEffect } from 'react';
import { useAuth0 } from '@auth0/auth0-react';

// ...

const App: React.FC = () => {
  // Récupérer le gestionnaire de rappel depuis le hook React d'Auth0
  const { handleRedirectCallback } = useAuth0();

  useEffect(() => {
    // Traiter l'événement 'appUrlOpen' et appeler `handleRedirectCallback`
    CapApp.addListener('appUrlOpen', async ({ url }) => {
      if (url.includes('state') && (url.includes('code') || url.includes('error'))) {
        await handleRedirectCallback(url);
      }
      // Ne rien faire sur Android
      await Browser.close();
    });
  }, [handleRedirectCallback]);

  // ..
