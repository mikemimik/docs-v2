server {
    include conf.d/openid_connect.server_conf; # Flux de code d'autorisation et traitement de la partie utilisatrice
    error_log /var/log/nginx/error.log debug;  # Réduire le niveau de gravité au besoin

    listen 8010; # Utiliser SSL/TLS en production

    location / {
        # Ce site est protégé avec OpenID Connect
        auth_jwt "" token=$session_jwt;
        error_page 401 = @do_oidc_flow;

        #auth_jwt_key_file $oidc_jwt_keyfile; # ファイル名を使用する場合に有効化
        auth_jwt_key_request /_jwks_uri; # URLを使用する場合に有効化

        # Les utilisateurs authentifiés avec succès sont acheminés par proxy vers le backend,
        # avec le claim 'sub' transmis comme en-tête HTTP
        proxy_set_header username $jwt_claim_sub;
        proxy_set_header x-email $jwt_claim_email;
        #proxy_set_header x-custom $claim_name;             # claim avec espace de noms

        proxy_pass http://my_backend; # Le site/application backend

        access_log /var/log/nginx/access.log main_jwt;
    }
}
