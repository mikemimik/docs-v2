```js login.js lines
<template>
  <div>
    <button @click="login">Log in</button>
  </div>
</template>
<script>
  // API de composition
  import { useAuth0 } from '@auth0/auth0-vue';

  export default {
    setup() {
      const auth0 = useAuth0();

      return {
        login() {
          auth0.loginWithRedirect();
        }
      };
    }
  };

  // API d'options
  export default {
    methods: {
      login() {
        this.$auth0.loginWithRedirect();
      }
    }
  };
</script>
```
