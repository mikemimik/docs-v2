// Importer les plugiciels d'application et de navigateur de Capacitor, nous donnant accès à `addListener` et `appUrlOpen`,
// ainsi que les éléments nécessaires pour Auth0 et React
import { App as CapApp } from '@capacitor/app';
import { Browser } from '@capacitor/browser';
import { useEffect } from 'react';
import { useAuth0 } from '@auth0/auth0-react';

// ...

const App: React.FC = () => {
  // Obtenir le gestionnaire de rappel depuis le Hook React Auth0
  const { handleRedirectCallback } = useAuth0();

  useEffect(() => {
    // Gérer l'événement 'appUrlOpen' et appeler `handleRedirectCallback`
    CapApp.addListener('appUrlOpen', async ({ url }) => {
      if (url.includes('state') && (url.includes('code') || url.includes('error'))) {
        await handleRedirectCallback(url);
      }
      // Sans effet sur Android
      await Browser.close();
    });
  }, [handleRedirectCallback]);

  // ..
