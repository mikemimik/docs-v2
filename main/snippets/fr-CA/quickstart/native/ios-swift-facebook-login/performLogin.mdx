private func performLogin(accessToken: AccessToken) {
    fetchSessionToken(accessToken: accessToken) { token, error in
        if let error = error {
            print(error)
        }
        if let token = token {
            self.fetchUserProfile() { userProfile, error in
                if let error = error {
                    print(error)
                }
                if let userProfile = userProfile {
                    self.exchangeToken(sessionToken: token, userProfile: userProfile) { credentials, error in
                        if let credentials = credentials {
                            let _ = self.credentialManager.store(credentials: credentials)
                        }
                    }
                }
            }
        }
    }
}
