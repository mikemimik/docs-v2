from authlib.integrations.django_oauth2 import ResourceProtector
from django.http import JsonResponse
from . import validator

require_auth = ResourceProtector()
validator = validator.Auth0JWTBearerTokenValidator(
    "{yourDomain}",
    "{yourApiIdentifier}"
)
require_auth.register_token_validator(validator)

def public(request):
    """No se requiere token de acceso para acceder a esta ruta
    """
    response = "Hello from a public endpoint! You don't need to be authenticated to see this."
    return JsonResponse(dict(message=response))

@require_auth(None)
def private(request):
    """Se requiere un token de acceso válido para acceder a esta ruta
    """
    response = "Hello from a private endpoint! You need to be authenticated to see this."
    return JsonResponse(dict(message=response))

@require_auth("read:messages")
def private_scoped(request):
    """Se requieren un token de acceso válido y un scope apropiado para acceder a esta ruta
    """
    response = "Hello from a private endpoint! You need to be authenticated and have a scope of read:messages to see this."
    return JsonResponse(dict(message=response))
