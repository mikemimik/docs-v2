package com.auth0.example.security;

    import org.springframework.context.annotation.Bean;
    import org.springframework.context.annotation.Configuration;
    import org.springframework.security.config.annotation.web.builders.HttpSecurity;
    import org.springframework.security.web.SecurityFilterChain;

    import static org.springframework.security.config.Customizer.withDefaults;

    /**
     * Configura nuestra aplicación con Spring Security para restringir el acceso a nuestros API endpoints.
     */
    @Configuration
    public class SecurityConfig {

@Bean
public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
    /*
    Aquí configuramos la seguridad necesaria para nuestros endpoints y preparamos nuestra app para que funcione como
    un OAuth2 Resource Server, usando validación JWT.
    */
    return http
            .authorizeHttpRequests((authorize) -> authorize
                .requestMatchers("/api/public").permitAll()
                .requestMatchers("/api/private").authenticated()
                .requestMatchers("/api/private-scoped").hasAuthority("SCOPE_read:messages")
            )
            .cors(withDefaults())
            .oauth2ResourceServer(oauth2 -> oauth2
                .jwt(withDefaults())
            )
            .build();
}
    }
