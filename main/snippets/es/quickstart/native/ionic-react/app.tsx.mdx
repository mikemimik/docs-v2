// Importa los plugins de app y browser de Capacitor, lo que nos da acceso a `addListener` y `appUrlOpen`,
// así como los elementos necesarios para Auth0 y React
import { App as CapApp } from '@capacitor/app';
import { Browser } from '@capacitor/browser';
import { useEffect } from 'react';
import { useAuth0 } from '@auth0/auth0-react';

// ...

const App: React.FC = () => {
  // Obtiene el controlador de devolución de llamada del hook de Auth0 para React
  const { handleRedirectCallback } = useAuth0();

  useEffect(() => {
    // Maneja el evento 'appUrlOpen' y llama a `handleRedirectCallback`
    CapApp.addListener('appUrlOpen', async ({ url }) => {
      if (url.includes('state') && (url.includes('code') || url.includes('error'))) {
        await handleRedirectCallback(url);
      }
      // Sin efecto en Android
      await Browser.close();
    });
  }, [handleRedirectCallback]);

  // ..
};// Importa los plugins de app y browser de Capacitor, lo que nos da acceso a `addListener` y `appUrlOpen`,
