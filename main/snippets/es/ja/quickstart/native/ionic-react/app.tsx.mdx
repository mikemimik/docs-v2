// Importa los plugins de aplicación y navegador de Capacitor para acceder a `addListener`, `appUrlOpen`,
// y todo lo necesario de Auth0 y React
import { App as CapApp } from '@capacitor/app';
import { Browser } from '@capacitor/browser';
import { useEffect } from 'react';
import { useAuth0 } from '@auth0/auth0-react';

// ...

const App: React.FC = () => {
  // Obtiene el manejador de devolución de llamada del hook de Auth0 React
  const { handleRedirectCallback } = useAuth0();

  useEffect(() => {
    // Maneja el evento 'appUrlOpen' e invoca `handleRedirectCallback`
    CapApp.addListener('appUrlOpen', async ({ url }) => {
      if (url.includes('state') && (url.includes('code') || url.includes('error'))) {
        await handleRedirectCallback(url);
      }
      // No hace nada en Android
      await Browser.close();
    });
  }, [handleRedirectCallback]);

  // ..
};// Importa los plugins de aplicación y navegador de Capacitor para acceder a `addListener`, `appUrlOpen`,
