---
title: "PasswordValidationResult"
---

```ts Example
export interface PasswordValidationResult {
  /**
   * すべてのパスワードポリシーチェックの総合的な結果。
   *
   * {@link PasswordComplexityRule} 内のすべてのルールの `status` が `"valid"` の場合にのみ `true`。
   */
  isValid: boolean;

  /**
   * 評価済みパスワードルールの配列。
   *
   * 各要素は {@link PasswordComplexityRule} であり、
   * {@link PasswordComplexityRule.status | status} フィールドには
   * そのルールのバリデーション結果が反映されます。
   */
  results: PasswordComplexityRule[];
}
```

<div id="properties">
  ## プロパティ
</div>

<ParamField body="isValid" type="boolean">
  すべてのパスワードポリシーチェックの全体的な結果です。

  [PasswordComplexityRule](/docs/ja/libraries/acul/js-sdk/Screens/interfaces/PasswordComplexityRule) 内のすべてのルールの `status` が `"valid"` と等しい場合にのみ `true` になります。
</ParamField>

<ParamField body="results" type={<span><a href="/docs/ja/libraries/acul/js-sdk/Screens/interfaces/PasswordComplexityRule">PasswordComplexityRule</a></span>}>
  評価されたパスワードルールの配列です。

  各項目は [PasswordComplexityRule](/docs/ja/libraries/acul/js-sdk/Screens/interfaces/PasswordComplexityRule) であり、その
  [status](/docs/ja/libraries/acul/js-sdk/Screens/interfaces/PasswordComplexityRule#status) フィールドには、そのルールについて算出された
  バリデーション結果が反映されます。
</ParamField>
