export interface CustomizedConsentMembers extends BaseMembers {
  /**
   * Customized Consentスクリーンの特定のプロパティとデータへのアクセスを提供します。
   * リクエストされている `scopes` と `authorizationDetails` のリストを含みます。
   * @type {ScreenMembersOnCustomizedConsent}
   */
  screen: ScreenMembersOnCustomizedConsent;

  /**
   * リクエストされたpermissionsと認可の詳細を承認（グラント）するユーザーの決定を送信します。
   * このactionは `action: "accept"` を指定して `/u/customized-consent` エンドポイントにPOSTします。
   * transactionのstateは自動的に含まれます。
   *
   * @param {CustomOptions} [payload] - 省略可能。requestと共に送信するカスタムキーと値のペアを格納するオブジェクト。
   *                                    これらのParametersはサーバーに送信されるフォームデータに含まれます。
   * @returns {Promise<void>} acceptのactionが正常に送信されると解決されるPromise。
   *                          成功した場合、Auth0は通常ユーザーをアプリケーションへ、
   *                          または認証フローの次のステップへリダイレクトします。
   * @throws {Error} フォームの送信に失敗した場合（例：ネットワークエラー、無効なstate）にエラーをスローします。
   *                 サーバー側のエラー（"invalid_request" など）は、JavaScriptエラーとしてスローされるのではなく、
   *                 操作後に `this.transaction.errors` に反映されます。
   *
   * @example
   * ```typescript
   * // 'customizedConsentManager' は CustomizedConsent SDK クラスのインスタンスであると仮定します
   * try {
   *   await customizedConsentManager.accept();
   *   // 成功した場合、ページは通常リダイレクトされます。
   * } catch (error) {
   *   // 送信中に発生した予期しないエラーを処理します。
   *   console.error("Failed to submit consent acceptance:", error);
   * }
   * // awaitの後、サーバー側のバリデーションエラーについては customizedConsentManager.transaction.errors を確認してください。
   * ```
   */
  accept(payload?: CustomOptions): Promise<void>;

  /**
   * リクエストされたpermissionsと認可の詳細を拒否（却下）するユーザーの決定を送信します。
   * このactionは `action: "deny"` を指定して `/u/customized-consent` エンドポイントにPOSTします。
   * transactionのstateは自動的に含まれます。
   *
   * @param {CustomOptions} [payload] - 省略可能。requestと共に送信するカスタムキーと値のペアを格納するオブジェクト。
   *                                    これらのParametersはサーバーに送信されるフォームデータに含まれます。

   *                          成功した場合、Auth0は通常ユーザーをリダイレクトし、アクセス拒否メッセージを
   *                          表示するか、アプリケーションにエラーを返すことがあります。
   * @throws {Error} フォームの送信に失敗した場合（例：ネットワークエラー、無効なstate）にエラーをスローします。
   *                 サーバー側のエラーは `this.transaction.errors` に反映されます。
   *
   * @example
   * ```typescript
   * // 'customizedConsentManager' は CustomizedConsent SDK クラスのインスタンスであると仮定します
   * try {
   *   await customizedConsentManager.deny({ reason_code: "user_rejected_details" }); // カスタムオプションの例
   *   // 成功した場合、ページは通常リダイレクトされます。
   * } catch (error) {
   *   console.error("Failed to submit consent denial:", error);
   * }
   * // awaitの後、サーバー側のバリデーションエラーについては customizedConsentManager.transaction.errors を確認してください。
   * ```
   */
  deny(payload?: CustomOptions): Promise<void>;
}

<ParamField body="accept" type="Promise<void>">
  要求された権限と認可の詳細を受け入れる (グラントする) というユーザーの決定を送信します。
  このアクションは、`action: "accept"` を指定して `/u/customized-consent` エンドポイントに送信します。
  transaction の state は自動的に含まれます。

  accept アクションが正常に送信されると解決される Promise。
  成功すると、Auth0 は通常、ユーザーをアプリケーションへリダイレクトするか、
  認証フローの次のステップに進めます。

  #### 例外

  フォームの送信に失敗した場合 (例: ネットワークの問題、無効な state) にエラーをスローします。
  サーバー側のエラー (&quot;invalid&#95;request&quot; など) は、JavaScript エラーとしてスローされるのではなく、
  操作後に `this.transaction.errors` に反映されます。

  ```typescript Example
  // 'customizedConsentManager' が CustomizedConsent SDK クラスのインスタンスであると仮定します
  try {
    await customizedConsentManager.accept();
    // 成功した場合、通常はページがリダイレクトされます。
  } catch (error) {
    // 送信処理自体で発生した予期しないエラーを処理します。
    console.error("Failed to submit consent acceptance:", error);
  }
  // await の後で、サーバー側の検証に関する問題がないか customizedConsentManager.transaction.errors を確認します。
  ```

<ParamField body="deny" type="Promise<void>">
  要求された権限と認可の詳細を拒否するというユーザーの決定を送信します。
  このアクションは、`action: "deny"` を指定して `/u/customized-consent` エンドポイントに送信します。
  transaction の state は自動的に含まれます。

  deny アクションが正常に送信されると解決される Promise。
  成功すると、Auth0 は通常ユーザーをリダイレクトし、アクセス拒否メッセージを
  表示したり、アプリケーションにエラーを返したりする場合があります。

  #### 例外

  フォームの送信に失敗した場合 (例: ネットワークの問題、無効な state) にエラーをスローします。
  サーバー側のエラーは `this.transaction.errors` に反映されます。

  ```typescript Example

    await customizedConsentManager.deny({ reason_code: "user_rejected_details" }); // カスタムオプションの例
    // 成功した場合、通常はページがリダイレクトされます。
  } catch (error) {
    console.error("Failed to submit consent denial:", error);
  }
  // await の後で、サーバー側の検証に関する問題がないか customizedConsentManager.transaction.errors を確認します。
  ```
