---
title: "ContinueOTPOptions"
---

OTP チャレンジを続行するオプション

```ts Example
export interface ContinueOptions {
  /** ユーザーが入力したコード */
  code: string;
  /** ブラウザを記憶するかどうかを示す */
  rememberDevice?: boolean;
  /** その他のカスタムオプション */
  [key: string]: string | number | boolean | undefined;
}
```

<div id="properties">
  ## プロパティ
</div>

<ParamField body="code" type="string">
  ユーザーが入力したコード
</ParamField>

<ParamField body="rememberDevice?" type="boolean">
  このブラウザを記憶するかどうかを示します
</ParamField>
