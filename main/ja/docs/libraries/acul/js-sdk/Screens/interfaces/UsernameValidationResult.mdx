---
title: "UsernameValidationResult"
---

1 つ以上のルールに対してユーザー名をバリデーションした結果です。

```ts Example
export interface UsernameValidationResult {
  /**
   * ユーザー名がすべてのバリデーションルールを通過したかどうかを示します。
   */
  isValid: boolean;

  /**
   * 失敗した個々のルールを表す {@link UsernameValidationError} オブジェクトの配列です（存在する場合）。
   *
   * {@link UsernameValidationResult.isValid} が `true` の場合、この配列は空になります。
   */
  errors: UsernameValidationError[];
}
```
