---
title: "OrganizationPickerMembers"
---

organization-picker画面で使用できるメソッドとプロパティを定義するインターフェース

```ts Example
export interface OrganizationPickerMembers extends BaseMembers {
  /**
   * ユーザーの組織を選択します。
   * @param payload 組織IDを含むオプション。
   */
  selectOrganization(payload: { organization: string; state: string }): Promise<void>;

  /**
   * 組織の選択をスキップし、ユーザーの個人アカウントで続行します。
   * @param payload requestに含めるカスタムオプション（任意）。
   */
  skipOrganizationSelection(payload?: CustomOptions): Promise<void>;
}
```
