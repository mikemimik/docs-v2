export interface ResetPasswordMfaWebAuthnRoamingChallengeMembers extends BaseMembers {

  /**
 * `reset-password-mfa-webauthn-roaming-challenge` 画面固有のプロパティとデータへのアクセスを提供します。
 * WebAuthn の `publicKey` チャレンジオプションと `showRememberDevice` フラグを含みます。
 * @type {ScreenMembersOnResetPasswordMfaWebAuthnRoamingChallenge}
 */
  screen: ScreenMembersOnResetPasswordMfaWebAuthnRoamingChallenge;

  /**
   * WebAuthn アサーションプロセス（セキュリティキーチャレンジ）を開始します。
   * このメソッドは内部的に、`screen.publicKey` で提供されたチャレンジを使用して
   * `navigator.credentials.get()` を呼び出します。
   * セキュリティキーとの対話が成功すると、結果として得られた
   * `PublicKeyCredential` を `action: "default"` とともに Auth0 に送信します。
   *
   * @param {UseSecurityKeyOptions} [options] - 省略可能。操作のパラメータ。
   * `rememberDevice`（`screen.showRememberDevice` が true の場合）やその他のカスタムオプションなどを指定できます。
   * `response` フィールド（WebAuthn 資格情報）は SDK が内部的に処理します。
   * @returns {Promise<void>} 検証の試行が送信されたときに解決される Promise。
   * 操作が成功した場合、通常は Auth0 によってリダイレクトが行われます。
   * @throws {Error} `screen.publicKey` が存在しない場合、`navigator.credentials.get()` が
   * 失敗した場合（例：ユーザーによるキャンセル、認証器が見つからない場合など）、または Auth0 へのフォーム送信が失敗した場合にエラーをスローします。
   * これらのエラーをキャッチし、WebAuthn API 固有の問題を報告するために `showError()` の使用を検討することを推奨します。
   */
  useSecurityKey(options?: UseSecurityKeyOptions): Promise<void>;

  /**
   * クライアントサイドの WebAuthn API エラー（`navigator.credentials.get()` からのもの）を Auth0 に報告します。
   * このメソッドは、`useSecurityKey()`（または手動での `navigator.credentials.get()` 呼び出し）が
   * ブラウザ側の WebAuthn の問題により失敗した場合に使用します（例：ユーザーがプロンプトをキャンセルした場合の `NotAllowedError`、
   * 一致する認証器が見つからない場合の `NotFoundError`、またはタイムアウトなど）。
   * エラーの詳細を `action: "showError::{errorDetailsJsonString}"` と空の `response` とともに送信します。
   *
   * @param {ShowErrorOptions} options - WebAuthn API の DOMException から取得した `name` と `message` を持つ `error` オブジェクト、
   * 省略可能な `rememberDevice` フラグ、およびその他のカスタムオプションを含みます。
   * @returns {Promise<void>} エラーレポートが送信されたときに解決される Promise。
   * Auth0 はエラー情報とともにページを再レンダリングするか、リダイレクトする場合があります。
   * @throws {Error} フォームの送信が失敗した場合（例：ネットワークエラー、invalid state）にエラーをスローします。
   */
  showError(options: ShowErrorOptions): Promise<void>;

  /**
   * ユーザーが WebAuthn ローミングチャレンジをオプトアウトし、別の MFA方式を選択できるようにします。
   * このアクションは `action: "pick-authenticator"` を Auth0 に送信し、
   * ユーザーを MFA 認証要素選択画面に遷移させます。
   *
   * @param {TryAnotherMethodOptions} [options] - 省略可能。操作のパラメータ。
   * `rememberDevice`（`screen.showRememberDevice` が true の場合）やその他のカスタムオプションなどを指定できます。
   * @returns {Promise<void>} 'pick-authenticator' アクションが送信されたときに解決される Promise。
   * @throws {Error} フォームの送信が失敗した場合にエラーをスローします。
   */
  tryAnotherMethod(options?: TryAnotherMethodOptions): Promise<void>;
}

<ParamField body="showError" type="Promise<void>">
  client-side の WebAuthn API エラー (`navigator.credentials.get()` からのエラー) を Auth0 に報告します。
  このメソッドは、`useSecurityKey()` (または手動の `navigator.credentials.get()` 呼び出し) が
  ブラウザー側の WebAuthn の問題 (たとえば、ユーザーがプロンプトをキャンセルした場合の `NotAllowedError`、
  一致する認証器が見つからない場合の `NotFoundError`、またはタイムアウト) によって失敗したときに使用します。
  エラーの詳細は、`action: "showError::{errorDetailsJsonString}"` と空の `response` とともに送信されます。

  エラー報告の送信完了時に解決される promise です。
  Auth0 は、エラー情報を表示してページを再レンダリングするか、リダイレクトすることがあります。

  #### 例外

  フォーム送信に失敗した場合 (たとえば、network error や invalid state) にエラーをスローします。
