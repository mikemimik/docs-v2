メールアドレスを選択する際のオプション

```ts Example
export interface SelectMfaEmailOptions {
  /**
   * @param {number} index - 選択する電話番号のインデックス。
   *
   */
  index: number;
  /** カスタムオプション */
  [key: string]: string | number | boolean | undefined;
}
```
