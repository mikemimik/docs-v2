export interface MfaWebAuthnNotAvailableErrorMembers extends BaseMembers {
  /**
   * WebAuthnが利用できないため、ユーザーが別のMFA方式を選択できるようにします。
   * このActionはユーザーを認証器選択画面に誘導し、利用可能な/登録済みの他のMFA認証要素から選択できます。
   *
   * @param {CustomOptions} [payload] - リクエストとともに送信されるオプションのカスタムパラメータ。
   * @returns {Promise<void>} 'pick-authenticator' Actionの送信が成功したときに解決されるPromise。
   * @throws {Error} フォームの送信が失敗した場合（例：ネットワークエラー、サーバー側のバリデーションエラー）にエラーをスローします。
   * @example
   * ```typescript
   * // 'sdk' は MfaWebAuthnNotAvailableError のインスタンスであると仮定します
   * try {
   *   await sdk.tryAnotherMethod();
   *   // 成功時、Auth0が認証器選択画面へのリダイレクトを処理します。
   * } catch (error) {
   *   console.error('Failed to switch to another MFA method:', error);
   *   // 必要に応じて、失敗をユーザーに通知するためUIを更新します。
   * }
   * ```
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;
}

<ParamField body="tryAnotherMethod" type="Promise<void>">
  WebAuthn が利用できない場合に、ユーザーが別の MFA方式 を選択できるようにします。
  このアクションにより、ユーザーは認証器選択画面に移動し、
  利用可能な、または登録済みの別の MFA 認証要素を選択できます。

  `pick-authenticator` Action の送信に成功すると解決される Promise です。

  #### 例外

  フォームの送信が失敗した場合 (例: ネットワークの問題、サーバー側のバリデーションエラー) にエラーをスローします。

  ```typescript Example
  // 'sdk' は MfaWebAuthnNotAvailableError のインスタンスであるとします
  try {
    await sdk.tryAnotherMethod();
    // 成功時は、Auth0 が認証器選択画面へのリダイレクトを処理します。
  } catch (error) {
    console.error('Failed to switch to another MFA method:', error);
    // 必要に応じて、失敗したことをユーザーに知らせるために UI を更新します。
  }
  ```
