title: "ResetPasswordMfaVoiceChallengeMembers"

reset-password-mfa-voice-challenge 画面で使用できるメソッドとプロパティを定義するインターフェース。

````ts Example
export interface ResetPasswordMfaVoiceChallengeMembers extends BaseMembers {
  screen: ScreenMembersOnResetPasswordMfaVoiceChallenge;

  /**
   * 指定されたコードを使用して音声チャレンジを続行します。
   *
   * @param payload - コードを含むオプション。
   * @returns challengeが送信されたときに解決されるPromise。
   *
   * @example
   * ```ts
   * const reset = new ResetPasswordMfaVoiceChallenge();
   * await reset.continue({ code: '123456' });
   * ```
   */
  continue(payload: ContinueOptions): Promise<void>;

  /**
   * SMS認証に切り替えます。
   *
   * @param payload - requestに含めるカスタムオプション（任意）。
   * @returns アクションが完了したときに解決されるPromise。
   */
  switchToSms(payload?: CustomOptions): Promise<void>;

  /**
   * voice callでコードを再送します。
   *
   * @param payload - requestに含めるカスタムオプション（任意）。
   * @returns コードが再送されたときに解決されるPromise。
   */
  resendCode(payload?: CustomOptions): Promise<void>;

  /**
   * ユーザーが別のMFA方式を試せるようにします。
   *
   * @param payload - requestに含めるカスタムオプション（任意）。
   * @returns requestが送信されたときに解決されるPromise。
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;

  /**
   * この画面のtimeout管理付き再送機能を取得します
   * @param options 再送機能の設定オプション
   */
  resendManager(options?: StartResendOptions): ResendControl;
}
````
