````ts Example
export interface MfaPhoneChallengeMembers extends BaseMembers {
  /**
   * @property {ScreenMembersOnMfaPhoneChallenge} screen - MFA電話チャレンジの特定の画面情報。
   */
  screen: ScreenMembersOnMfaPhoneChallenge;

  /**
   * continue
   * チャレンジを続行するために、MFAの配信方法（SMSまたは音声）の選択を送信します。
   * これにより、ユーザーの電話に確認コードが送信されます。
   *
   * @param {ContinueOptions} payload - 配信タイプ（'sms' または 'voice'）を指定するオプション。
   * @returns {Promise<void>} アクションの送信が完了すると解決されるPromise。
   * @throws {Error} 送信に失敗した場合にエラーをスローします。
   * @example
   * ```typescript
   * import MfaPhoneChallenge from '@auth0/auth0-acul-js/mfa-phone-challenge';
   *
   * const mfaPhoneChallenge = new MfaPhoneChallenge();
   *
   * // SMSで確認コードを送信する場合
   * await mfaPhoneChallenge.continue({ type: 'sms' });
   *
   * // 音声通話で確認コードを送信する場合
   * await mfaPhoneChallenge.continue({ type: 'voice' });
   * ```
   */
  continue(payload: ContinueOptions): Promise<void>;

  /**
   * pickPhone
   * 別の電話番号を選択するアクションを開始します。
   * 該当する場合、登録済みの複数の電話番号から選択できる画面にユーザーをリダイレクトします。
   *
   * @param {PickPhoneOptions} [payload] - リクエストに含めるカスタムオプション（省略可能）。
   * @returns {Promise<void>} アクションの送信が完了すると解決されるPromise。
   * @throws {Error} 送信に失敗した場合にエラーをスローします。
   * @example
   * ```typescript
   * import MfaPhoneChallenge from '@auth0/auth0-acul-js/mfa-phone-challenge';
   *
   * const mfaPhoneChallenge = new MfaPhoneChallenge();
   * await mfaPhoneChallenge.pickPhone();
   * ```
   */
  pickPhone(payload?: PickPhoneOptions): Promise<void>;

  /**
   * tryAnotherMethod
   * 別の認証要素/認証器を選択するアクションを開始します。
   * 代替MFA方式（例：OTPアプリ、リカバリーコード）を選択できる画面にユーザーをリダイレクトします。
   *
   * @param {PickAuthenticatorOptions} [payload] - リクエストに含めるカスタムオプション（省略可能）。
   * @returns {Promise<void>} アクションの送信が完了すると解決されるPromise。
   * @throws {Error} 送信に失敗した場合にエラーをスローします。
   * @example
   * ```typescript
   * import MfaPhoneChallenge from '@auth0/auth0-acul-js/mfa-phone-challenge';
   *
   * const mfaPhoneChallenge = new MfaPhoneChallenge();
   * await mfaPhoneChallenge.tryAnotherMethod();
   * ```
   */
  tryAnotherMethod(payload?: PickAuthenticatorOptions): Promise<void>;
}

<ParamField body="continue" type="Promise<void>">
  continue
  チャレンジを続行するために、MFAの配信方法 (SMS または Voice) を送信します。
  これにより、ユーザーの電話番号に確認コードが送信されます。

  アクションが送信されると解決される Promise。

  #### 例外

  送信に失敗した場合はエラーをスローします。

  ```typescript Example
  import MfaPhoneChallenge from '@auth0/auth0-acul-js/mfa-phone-challenge';

  const mfaPhoneChallenge = new MfaPhoneChallenge();

  // SMS でコードを送信する
  await mfaPhoneChallenge.continue({ type: 'sms' });

  // 音声通話 でコードを送信する
  await mfaPhoneChallenge.continue({ type: 'voice' });
  ```

  <Expandable title="パラメーター">
    <ParamField body="payload" type={<span><a href="/ja/docs/libraries/acul/react-sdk/API-Reference/Types/interfaces/MfaPhoneChallengeContinueOptions">MfaPhoneChallengeContinueOptions</a></span>}>
      配信タイプ (`sms` または `voice`) を指定するオプションです。

      **プロパティ**

      <ParamField body="type" type="&#x22;sms&#x22; | &#x22;voice&#x22;">
        実行するチャレンジの種類です。

        * `sms`: テキストメッセージでコードを送信します。
        * `voice`: 音声通話でコードを送信します。
      </ParamField>
    </ParamField>
  </Expandable>
</ParamField>
