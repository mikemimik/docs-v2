```ts Example
export interface ShowErrorOptions {
  /**
   * WebAuthnエラーの詳細。
   */

  /**
   * その他のカスタムオプション。
   */
  [key: string]: string | number | boolean | undefined | WebAuthnErrorDetails;
}
```
