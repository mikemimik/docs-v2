export interface ResetPasswordMfaWebAuthnRoamingChallengeMembers extends BaseMembers {

  /**
 * `reset-password-mfa-webauthn-roaming-challenge` 画面の特定のプロパティとデータへのアクセスを提供します。
 * WebAuthn の `publicKey` チャレンジオプションと `showRememberDevice` フラグを含みます。
 * @type {ScreenMembersOnResetPasswordMfaWebAuthnRoamingChallenge}
 */

  /**

   * このメソッドは内部的に、`screen.publicKey` で提供されたチャレンジを使用して
   * `navigator.credentials.get()` を呼び出します。
   * セキュリティキーとの対話が成功すると、結果として得られた
   * `PublicKeyCredential` を `action: "default"` とともに Auth0 に送信します。
   *

   * `response` フィールド（WebAuthn 資格情報）は SDK が内部的に処理します。
   * @returns {Promise<void>} 検証の試行が送信されたときに解決される Promise。
   * 操作が成功した場合、通常は Auth0 によってリダイレクトが行われます。

   * これらのエラーをキャッチし、WebAuthn API 固有の問題を報告するために `showError()` の使用を検討することを推奨します。
   */

  /**
   * クライアントサイドの WebAuthn API エラー（`navigator.credentials.get()` からのもの）を Auth0 に報告します。
   * このメソッドは、`useSecurityKey()`（または手動の `navigator.credentials.get()` 呼び出し）が

   * 省略可能な `rememberDevice` フラグ、およびその他のカスタムオプションを含みます。

   */
  showError(options: ShowErrorOptions): Promise<void>;

  /**
   * ユーザーが WebAuthn ローミングチャレンジをオプトアウトし、別の MFA方式を選択できるようにします。
   * このアクションは `action: "pick-authenticator"` を Auth0 に送信し、
   * ユーザーを認証要素の選択画面に誘導します。
   *
   * @param {TryAnotherMethodOptions} [options] - 省略可能。操作のパラメータ。
   * `rememberDevice`（`screen.showRememberDevice` が true の場合）やその他のカスタムオプションなどを指定できます。
   * @returns {Promise<void>} 'pick-authenticator' アクションが送信されたときに解決される Promise。
   * @throws {Error} フォームの送信が失敗した場合にエラーをスローします。
   */
  tryAnotherMethod(options?: TryAnotherMethodOptions): Promise<void>;
}

<ParamField body="showError" type="Promise<void>">
  クライアント側の WebAuthn API エラー (`navigator.credentials.get()` によるもの) を Auth0 に報告します。
  このメソッドは、`useSecurityKey()` (または `navigator.credentials.get()` の手動呼び出し) が、
  ブラウザー側の WebAuthn の問題 (たとえば、ユーザーがプロンプトをキャンセルした場合の `NotAllowedError`、
  一致する認証器が見つからない場合の `NotFoundError`、またはタイムアウト) によって失敗したときに使用してください。
  `action: "showError::{errorDetailsJsonString}"` と空の `response` を付けて、エラーの詳細を送信します。

  エラー報告の送信が完了すると解決される Promise。
  Auth0 は、エラー情報を表示するためにページを再レンダリングするか、リダイレクトすることがあります。

  #### 例外

  Form の送信に失敗した場合 (たとえば、ネットワークエラーや `invalid state`) はエラーをスローします。

<ParamField body="useSecurityKey" type="Promise<void>">
  WebAuthn のアサーションプロセス (セキュリティキーのチャレンジ) を開始します。
  このメソッドは内部で、`screen.publicKey` で提供された challenge を使って
  `navigator.credentials.get()` を呼び出します。
  セキュリティキーとのやり取りに成功すると、生成された
  `PublicKeyCredential` を `action: "default"` とともに Auth0 に送信します。

  検証の試行が送信されると解決される Promise です。
  通常、操作が成功すると Auth0 によってリダイレクトされます。

  #### Throws

  `screen.publicKey` がない場合、`navigator.credentials.get()`
  が失敗した場合 (例: ユーザーによるキャンセル、認証器が見つからない場合) 、または Auth0 へのフォーム送信が失敗した場合にエラーをスローします。
  これらのエラーは catch し、必要に応じて `showError()` を使って WebAuthn API 固有の問題を表示することをおすすめします。
