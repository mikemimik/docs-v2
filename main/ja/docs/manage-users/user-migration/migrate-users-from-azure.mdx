title: Azure からユーザーを移行する
---

Auth0 では、Auth0 の[カスタム DB の Login スクリプト](/docs/ja/authenticate/database-connections/custom-db/templates/login)で [Azure AD ROPC フロー](https://docs.microsoft.com/en-us/azure/active-directory/develop/v2-oauth-ropc)を使用して、Azure からユーザーをインポートできます。

1. 各 tenant またはディレクトリで、**公開クライアント**として [公開クライアントのアプリ登録を作成](https://docs.microsoft.com/en-us/azure/active-directory/develop/quickstart-register-app)し、API 呼び出し用に提供される Client ID と Client Secret を安全に保管します。
2. このアプリケーションを使用できるユーザー、またはこの API にアクセスできるユーザーを尋ねられたら、**任意の組織ディレクトリ内のアカウント**を選択します。
3. Microsoft Graph セクションで、`Directory.Read.All` と `User.Read` に権限を付与します。
4. アプリが B2C の場合は、[Azure Active Directory B2C でリソース所有者パスワード資格情報フローを設定](https://learn.microsoft.com/en-us/azure/active-directory-b2c/add-ropc-policy?tabs=app-reg-ga\&pivots=b2c-user-flow)します。

これで ROPC フローを使用できます。以下のスニペットは呼び出し例です。

<div id="azure-ad-login-script-template">
  ##### Azure AD Login スクリプト テンプレート:
</div>

```javascript lines expandable
function login(email, password, callback) {
      const axios = require('axios').default;
      const qs = require('qs');
    	const jwtDecode = require('jwt-decode').default;

       const data = { 'client_id': {{YOUR_AZURE_APP_CLIENT_ID}},
             'scope': 'user.read openid profile offline_access',
             'username': email,
             'password': password,
             'grant_type': 'password',
             'client_secret': {{YOUR_AZURE_APP_CLIENT_SECRET}}
             };
const options = {
  method: 'POST',
  headers: { 'content-type': 'application/x-www-form-urlencoded' },
  data: qs.stringify(data),
  url: 'https://login.microsoftonline.com/{{AZURE_TENANT_ID}}/oauth2/v2.0/token',
};

axios(options).then(
	response => {
  // 本番環境では、クレームを読み取る前にこのJWTを完全に検証することをお勧めします
  const claims = jwtDecode(response.data.id_token);

  if (response.statusCode === 401) return callback();

  callback(null, {
  // これはAuth0ユーザープロファイルにマッピングできるプロパティの簡単な例です
  // Azureトークンから取得したデータに基づいて、マッピングする内容を自由に選択できます
  user_id: claims.sub,
  nickname: claims.name,
  email: claims.preferred_username
});

}
      );

    }
```

```javascript lines expandable
function login(email, password, callback) {
      const axios = require('axios').default;
      const qs = require('qs');
    	const jwtDecode = require('jwt-decode').default;

       const data = { 'client_id': {{YOUR_B2C_APP_CLIENT_ID}},
             'scope': 'openid {{YOUR_B2C_APP_CLIENT_ID}} offline_access profile',
             'username': email,
             'password': password,
             'grant_type': 'password',
             };
const options = {
  method: 'POST',
  headers: { 'content-type': 'application/x-www-form-urlencoded' },
  data: qs.stringify(data),
  url: 'https://{{TENANT_NAME}}.b2clogin.com/{{TENANT_NAME}}.onmicrosoft.com/{{B2C_POLICY_NAME}}/oauth2/v2.0/token',
};

axios(options).then(
	response => {
  // 本番環境では、クレームを読み取る前にこのJWTを完全に検証することをお勧めします
  const claims = jwtDecode(response.data.access_token);
  if (response.statusCode === 401) return callback();

  callback(null, {
  // これはAuth0 ユーザープロファイルにマッピングできるプロパティの簡単な例です
  // azureトークンから取得したデータに基づいて、マッピングする内容を自由に選択できます
  user_id: claims.sub,
  nickname: claims.given_name,
  email: claims.emails[0]
});
}
      );
    }
```
